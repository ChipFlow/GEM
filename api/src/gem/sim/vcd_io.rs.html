19/// Hierarchical name representation in VCD.
20#[derive(PartialEq, Eq, Clone, Debug)]
21pub struct VCDHier {
22    pub cur: CompactString,
23    pub prev: Option<Rc<VCDHier>>,
24}
25
26/// Reverse iterator of a [`VCDHier`], yielding cell names
27/// from the bottom to the top module.
28pub struct VCDHierRevIter<'i>(Option<&'i VCDHier>);
29
30impl<'i> Iterator for VCDHierRevIter<'i> {
31    type Item = &'i CompactString;
32
33    #[inline]
34    fn next(&mut self) -> Option<&'i CompactString> {
35        let name = self.0?;
36        if name.cur.is_empty() {
37            return None;
38        }
39        let ret = &name.cur;
40        self.0 = name.prev.as_ref().map(|a| a.as_ref());
41        Some(ret)
42    }
43}
44
45impl<'i> IntoIterator for &'i VCDHier {
46    type Item = &'i CompactString;
47    type IntoIter = VCDHierRevIter<'i>;
48
49    #[inline]
50    fn into_iter(self) -> VCDHierRevIter<'i> {
51        VCDHierRevIter(Some(self))
52    }
53}
54
55impl Hash for VCDHier {
56    fn hash<H: std::hash::Hasher>(&self, state: &mut H) {
57        for s in self.iter() {
58            s.hash(state);
59        }
60    }
61}
62
63#[allow(dead_code)]
64impl VCDHier {
65    #[inline]
66    pub fn single(cur: CompactString) -> Self {
67        VCDHier { cur, prev: None }
68    }
69
70    #[inline]
71    pub fn empty() -> Self {
72        VCDHier {
73            cur: "".into(),
74            prev: None,
75        }
76    }
77
78    #[inline]
79    pub fn is_empty(&self) -> bool {
80        self.cur.as_str() == "" && self.prev.is_none()
81    }
82
83    #[inline]
84    pub fn iter(&self) -> VCDHierRevIter<'_> {
85        (&self).into_iter()
86    }
87}

91/// Try to match one component in a scope path.
92/// Returns the remaining scope on success, or None on failure.
93pub fn match_scope_path<'i>(mut scope: &'i str, cur: &str) -> Option<&'i str> {
94    if scope.is_empty() {
95        return Some("");
96    }
97    if scope.starts_with('/') {
98        scope = &scope[1..];
99    }
100    if scope.is_empty() {
101        Some("")
102    } else if scope.starts_with(cur) {
103        if scope.len() == cur.len() {
104            Some("")
105        } else if scope.as_bytes()[cur.len()] == b'/' {
106            Some(&scope[cur.len() + 1..])
107        } else {
108            None
109        }
110    } else {
111        None
112    }
113}

115/// Find a scope by its path in the VCD hierarchy.
116pub fn find_top_scope<'i>(items: &'i [ScopeItem], top_scope: &'_ str) -> Option<&'i Scope> {
117    for item in items {
118        if let ScopeItem::Scope(scope) = item {
119            if let Some(s1) = match_scope_path(top_scope, scope.identifier.as_str()) {
120                return match s1 {
121                    "" => Some(scope),
122                    _ => find_top_scope(&scope.children[..], s1),
123                };
124            }
125        }
126    }
127    None
128}

130/// Recursively collect all scope paths from VCD header.
131pub fn collect_all_scopes<'a>(
132    items: &'a [ScopeItem],
133    prefix: &str,
134    scopes: &mut Vec<(String, &'a Scope)>,
135) {
136    for item in items {
137        if let ScopeItem::Scope(scope) = item {
138            let scope_path = if prefix.is_empty() {
139                scope.identifier.to_string()
140            } else {
141                format!("{}/{}", prefix, scope.identifier)
142            };
143            scopes.push((scope_path.clone(), scope));
144            collect_all_scopes(&scope.children[..], &scope_path, scopes);
145        }
146    }
147}

149/// Get required input port names from netlistdb.
150pub fn get_required_input_ports(netlistdb: &NetlistDB) -> HashSet<String> {
151    let mut ports = HashSet::new();
152    for i in netlistdb.cell2pin.iter_set(0) {
153        if netlistdb.pindirect[i] != Direction::I {
154            let port_name = netlistdb.pinnames[i].1.to_string();
155            ports.insert(port_name);
156        }
157    }
158    ports
159}

161/// Check if a VCD scope contains all required ports.
162pub fn check_scope_contains_ports(scope: &Scope, required_ports: &HashSet<String>) -> bool {
163    let mut found_ports = HashSet::new();
164    for item in &scope.children {
165        if let ScopeItem::Var(var) = item {
166            found_ports.insert(var.reference.to_string());
167        }
168    }
169
170    for port in required_ports {
171        if !found_ports.contains(port) {
172            return false;
173        }
174    }
175    true
176}

178/// Auto-detect VCD scope containing the DUT.
179pub fn auto_detect_vcd_scope<'i>(
180    items: &'i [ScopeItem],
181    netlistdb: &NetlistDB,
182    top_module_name: &str,
183) -> Option<(String, &'i Scope)> {
184    let required_ports = get_required_input_ports(netlistdb);
185
186    if required_ports.is_empty() {
187        clilog::warn!("No input ports found in design - cannot auto-detect VCD scope");
188        return None;
189    }
190
191    let mut all_scopes = Vec::new();
192    collect_all_scopes(items, "", &mut all_scopes);
193
194    if all_scopes.is_empty() {
195        clilog::warn!("No scopes found in VCD file");
196        return None;
197    }
198
199    clilog::debug!(
200        "Searching for VCD scope containing {} input ports",
201        required_ports.len()
202    );
203    clilog::debug!("Required ports: {:?}", required_ports);
204
205    // Try common DUT scope names first
206    let common_names = ["dut", "uut", "DUT", "UUT", top_module_name];
207    for name in &common_names {
208        for (path, scope) in &all_scopes {
209            if path.ends_with(name) && check_scope_contains_ports(scope, &required_ports) {
210                clilog::info!(
211                    "Auto-detected VCD scope: {} (matched common pattern '{}')",
212                    path,
213                    name
214                );
215                return Some((path.clone(), *scope));
216            }
217        }
218    }
219
220    // Try any scope that contains all required ports
221    for (path, scope) in &all_scopes {
222        if check_scope_contains_ports(scope, &required_ports) {
223            clilog::info!(
224                "Auto-detected VCD scope: {} (contains all required ports)",
225                path
226            );
227            return Some((path.clone(), *scope));
228        }
229    }
230
231    clilog::error!("Could not auto-detect VCD scope. Available scopes:");
232    for (path, _) in &all_scopes {
233        clilog::error!("  - {}", path);
234    }
235    clilog::error!("Please specify scope manually with --input-vcd-scope");
236    None
237}

241/// Result of parsing input VCD: the state vectors and cycle timestamps.
242pub struct ParsedInputVCD {
243    /// Flattened input state vectors, one per cycle + trailing state.
244    pub input_states: Vec<u32>,
245    /// (offset_into_input_states, vcd_timestamp) per cycle.
246    pub offsets_timestamps: Vec<(usize, u64)>,
247}

249/// Resolve the top scope from VCD header, either from user-specified path or auto-detection.
250pub fn resolve_vcd_scope<'i>(
251    header_items: &'i [ScopeItem],
252    input_vcd_scope: Option<&str>,
253    netlistdb: &NetlistDB,
254    top_module: Option<&str>,
255) -> &'i Scope {
256    if let Some(scope_path) = input_vcd_scope {
257        clilog::info!("Using user-specified VCD scope: {}", scope_path);
258        find_top_scope(header_items, scope_path).expect("Specified top scope not found in VCD.")
259    } else {
260        let top_module_name = top_module.unwrap_or("top");
261        clilog::info!("No VCD scope specified - attempting auto-detection");
262
263        match auto_detect_vcd_scope(header_items, netlistdb, top_module_name) {
264            Some((_path, scope)) => scope,
265            None => {
266                panic!(
267                    "Failed to auto-detect VCD scope. Please specify --input-vcd-scope manually."
268                );
269            }
270        }
271    }
272}

274/// Match VCD variables to netlist input ports.
275///
276/// Returns:
277/// - `vcd2inp`: maps (vcd_code, bit_position) → netlist pin ID
278/// - `inp_port_given`: set of netlist pin IDs that were found in VCD
279pub fn match_vcd_inputs(
280    top_scope: &Scope,
281    netlistdb: &NetlistDB,
282) -> (HashMap<(u64, usize), usize>, HashSet<usize>) {
283    use sverilogparse::SVerilogRange;
284    use vcd_ng::ReferenceIndex::*;
285
286    let mut vcd2inp = HashMap::new();
287    let mut inp_port_given = HashSet::new();
288
289    let mut match_one_input = |var: &Var, i: Option<isize>, vcd_pos: usize| {
290        let key = (VCDHier::empty(), var.reference.as_str(), i);
291        if let Some(&id) = netlistdb.pinname2id.get(&key as &dyn GeneralPinName) {
292            if netlistdb.pindirect[id] != Direction::O {
293                return;
294            }
295            vcd2inp.insert((var.code.0, vcd_pos), id);
296            inp_port_given.insert(id);
297        }
298    };
299
300    for scope_item in &top_scope.children[..] {
301        if let ScopeItem::Var(var) = scope_item {
302            match var.index {
303                None => match var.size {
304                    1 => match_one_input(var, None, 0),
305                    w => {
306                        for (pos, i) in (0..w).rev().enumerate() {
307                            match_one_input(var, Some(i as isize), pos)
308                        }
309                    }
310                },
311                Some(BitSelect(i)) => match_one_input(var, Some(i as isize), 0),
312                Some(Range(a, b)) => {
313                    for (pos, i) in SVerilogRange(a as isize, b as isize).enumerate() {
314                        match_one_input(var, Some(i), pos);
315                    }
316                }
317            }
318        }
319    }
320
321    // Warn about missing primary inputs
322    for i in netlistdb.cell2pin.iter_set(0) {
323        if netlistdb.pindirect[i] != Direction::I && !inp_port_given.contains(&i) {
324            clilog::warn!(
325                GATESIM_VCDI_MISSING_PI,
326                "Primary input port {:?} not present in the VCD input",
327                netlistdb.pinnames[i]
328            );
329        }
330    }
331
332    (vcd2inp, inp_port_given)
333}

335/// Parse input VCD flow into state vectors for simulation.
336pub fn parse_input_vcd(
337    vcdflow: &mut vcd_ng::FastFlow<std::fs::File>,
338    vcd2inp: &HashMap<(u64, usize), usize>,
339    aig: &AIG,
340    script: &FlattenedScriptV1,
341    netlistdb: &NetlistDB,
342    max_cycles: Option<usize>,
343) -> ParsedInputVCD {
344    use vcd_ng::{FFValueChange, FastFlowToken};
345
346    let mut state = vec![0; script.reg_io_state_size as usize];
347    let mut vcd_time_last_active = u64::MAX;
348    let mut vcd_time = 0;
349    let mut last_vcd_time_active = true;
350    let mut delayed_bit_changes = HashSet::new();
351
352    let mut input_states = Vec::new();
353    let mut offsets_timestamps = Vec::new();
354
355    while let Some(tok) = vcdflow.next_token().unwrap() {
356        match tok {
357            FastFlowToken::Timestamp(t) => {
358                if t == vcd_time {
359                    continue;
360                }
361                if last_vcd_time_active {
362                    input_states.extend(state.iter().copied());
363                    offsets_timestamps.push((input_states.len(), vcd_time_last_active));
364                    for (_, &(pe, ne)) in &aig.clock_pin2aigpins {
365                        if pe != usize::MAX {
366                            let p = *script.input_map.get(&pe).unwrap();
367                            state[p as usize >> 5] &= !(1 << (p & 31));
368                        }
369                        if ne != usize::MAX {
370                            let p = *script.input_map.get(&ne).unwrap();
371                            state[p as usize >> 5] &= !(1 << (p & 31));
372                        }
373                    }
374                    if let Some(max_cycles) = max_cycles {
375                        if offsets_timestamps.len() >= max_cycles {
376                            clilog::info!("reached maximum cycles, stop reading input vcd");
377                            break;
378                        }
379                    }
380                }
381                if last_vcd_time_active {
382                    vcd_time_last_active = vcd_time;
383                }
384                vcd_time = t;
385                last_vcd_time_active = false;
386
387                for pos in std::mem::take(&mut delayed_bit_changes) {
388                    state[(pos >> 5) as usize] ^= 1u32 << (pos & 31);
389                }
390            }
391            FastFlowToken::Value(FFValueChange { id, bits }) => {
392                for (pos, b) in bits.iter().enumerate() {
393                    if let Some(&pin) = vcd2inp.get(&(id.0, pos)) {
394                        let aigpin = aig.pin2aigpin_iv[pin];
395                        assert_eq!(aigpin & 1, 0);
396                        let aigpin = aigpin >> 1;
397                        let pos = match script.input_map.get(&aigpin).copied() {
398                            Some(pos) => pos,
399                            None => {
400                                panic!(
401                                    "input pin {:?} (netlist id {}, aigpin {}) not found in output map.",
402                                    netlistdb.pinnames[pin].dbg_fmt_pin(),
403                                    pin,
404                                    aigpin
405                                );
406                            }
407                        };
408                        let old_value = state[(pos >> 5) as usize] >> (pos & 31) & 1;
409                        if old_value
410                            == match b {
411                                b'1' => 1,
412                                _ => 0,
413                            }
414                        {
415                            continue;
416                        }
417                        if let Some((pe, ne)) = aig.clock_pin2aigpins.get(&pin).copied() {
418                            if pe != usize::MAX && old_value == 0 {
419                                last_vcd_time_active = true;
420                                let p = *script.input_map.get(&pe).unwrap();
421                                state[p as usize >> 5] |= 1 << (p & 31);
422                            }
423                            if ne != usize::MAX && old_value == 1 {
424                                last_vcd_time_active = true;
425                                let p = *script.input_map.get(&ne).unwrap();
426                                state[p as usize >> 5] |= 1 << (p & 31);
427                            }
428                            delayed_bit_changes.insert(pos);
429                        } else {
430                            state[(pos >> 5) as usize] ^= 1u32 << (pos & 31);
431                        }
432                    }
433                }
434            }
435        }
436    }
437    input_states.extend(state.iter().copied());
438    clilog::info!("total number of cycles: {}", offsets_timestamps.len());
439
440    ParsedInputVCD {
441        input_states,
442        offsets_timestamps,
443    }
444}

446// ── X-propagation state buffer helpers ───────────────────────────────────────
447
448/// Expand value-only state buffer into a doubled buffer with X-mask for GPU dispatch.
449///
450/// The input `value_states` has `(num_cycles + 1)` snapshots of `reg_io_state_size` words.
451/// The output has `(num_cycles + 1)` snapshots of `2 * reg_io_state_size` words:
452/// `[values | xmask]` where X-mask is 0xFFFFFFFF for DFF output positions (unknown)
453/// and 0 for primary input positions (known from VCD).
454pub fn expand_states_for_xprop(
455    value_states: &[u32],
456    script: &FlattenedScriptV1,
457) -> Vec<u32> {
458    let rio = script.reg_io_state_size as usize;
459    let num_snapshots = value_states.len() / rio;
460    assert_eq!(value_states.len(), num_snapshots * rio);
461
462    // Build X-mask template: 0xFFFFFFFF for DFF positions, 0 for inputs
463    let mut xmask_template = vec![0xFFFF_FFFFu32; rio];
464    for &pos in script.input_map.values() {
465        xmask_template[(pos >> 5) as usize] &= !(1u32 << (pos & 31));
466    }
467
468    let mut expanded = Vec::with_capacity(num_snapshots * rio * 2);
469    for snap_i in 0..num_snapshots {
470        // Value portion
471        expanded.extend_from_slice(&value_states[snap_i * rio..(snap_i + 1) * rio]);
472        // X-mask portion
473        expanded.extend_from_slice(&xmask_template);
474    }
475    expanded
476}

478/// Extract value-only states from a doubled GPU output buffer.
479///
480/// Input has `(num_cycles + 1)` snapshots of `2 * reg_io_state_size` words.
481/// Returns separate value and X-mask arrays, each with the same layout as the
482/// original value-only state buffer (indexed by `offsets_timestamps` offsets).
483pub fn split_xprop_states(
484    gpu_states: &[u32],
485    reg_io_state_size: usize,
486) -> (Vec<u32>, Vec<u32>) {
487    let eff = reg_io_state_size * 2;
488    let num_snapshots = gpu_states.len() / eff;
489    assert_eq!(gpu_states.len(), num_snapshots * eff);
490
491    let mut values = Vec::with_capacity(num_snapshots * reg_io_state_size);
492    let mut xmasks = Vec::with_capacity(num_snapshots * reg_io_state_size);
493    for snap_i in 0..num_snapshots {
494        let base = snap_i * eff;
495        values.extend_from_slice(&gpu_states[base..base + reg_io_state_size]);
496        xmasks.extend_from_slice(&gpu_states[base + reg_io_state_size..base + eff]);
497    }
498    (values, xmasks)
499}

509/// Set up output VCD writer: add wire definitions and build output mapping.
510pub fn setup_output_vcd(
511    writer: &mut vcd_ng::Writer<std::io::BufWriter<std::fs::File>>,
512    header: &vcd_ng::Header,
513    output_vcd_scope: Option<&str>,
514    netlistdb: &NetlistDB,
515    aig: &AIG,
516    script: &FlattenedScriptV1,
517) -> OutputVCDMapping {
518    use vcd_ng::SimulationCommand;
519
520    if let Some((ratio, unit)) = header.timescale {
521        writer.timescale(ratio, unit).unwrap();
522    }
523    let output_vcd_scope_str = output_vcd_scope.unwrap_or("gem_top_module");
524    let scope_parts = output_vcd_scope_str.split('/').collect::<Vec<_>>();
525    for &scope in &scope_parts {
526        writer.add_module(scope).unwrap();
527    }
528
529    let out2vcd = netlistdb
530        .cell2pin
531        .iter_set(0)
532        .filter_map(|i| {
533            if netlistdb.pindirect[i] == Direction::I {
534                let aigpin = aig.pin2aigpin_iv[i];
535                if matches!(aig.drivers[aigpin >> 1], DriverType::InputPort(_)) {
536                    clilog::info!(
537                        "skipped output for port {} as it is a pass-through of input port.",
538                        netlistdb.pinnames[i].dbg_fmt_pin()
539                    );
540                    return None;
541                }
542                if aigpin <= 1 {
543                    return Some((
544                        aigpin,
545                        u32::MAX,
546                        writer
547                            .add_wire(1, &format!("{}", netlistdb.pinnames[i].dbg_fmt_pin()))
548                            .unwrap(),
549                    ));
550                }
551                Some((
552                    aigpin,
553                    *script.output_map.get(&aigpin).unwrap(),
554                    writer
555                        .add_wire(1, &format!("{}", netlistdb.pinnames[i].dbg_fmt_pin()))
556                        .unwrap(),
557                ))
558            } else {
559                None
560            }
561        })
562        .collect::<Vec<_>>();
563
564    for _ in 0..scope_parts.len() {
565        writer.upscope().unwrap();
566    }
567    writer.enddefinitions().unwrap();
568    writer.begin(SimulationCommand::Dumpvars).unwrap();
569
570    OutputVCDMapping { out2vcd }
571}

597fn write_output_vcd_impl(
598    writer: &mut vcd_ng::Writer<std::io::BufWriter<std::fs::File>>,
599    output_mapping: &OutputVCDMapping,
600    offsets_timestamps: &[(usize, u64)],
601    states: &[u32],
602    xmask_states: Option<&[u32]>,
603) {
604    use vcd_ng::Value;
605
606    clilog::info!("write out vcd");
607    // Use 3 to represent X in the last_val tracker (0=V0, 1=V1, 2=initial, 3=X)
608    let mut last_val = vec![2u32; output_mapping.out2vcd.len()];
609    let mut x_output_count = 0u64;
610    for &(offset, timestamp) in offsets_timestamps {
611        if timestamp == u64::MAX {
612            continue;
613        }
614        writer.timestamp(timestamp).unwrap();
615        for (i, &(output_aigpin, output_pos, vid)) in output_mapping.out2vcd.iter().enumerate() {
616            let (value_new, is_x) = match output_pos {
617                u32::MAX => {
618                    assert!(output_aigpin <= 1);
619                    (output_aigpin as u32, false)
620                }
621                output_pos => {
622                    let v = states[offset + (output_pos >> 5) as usize] >> (output_pos & 31) & 1;
623                    let x = xmask_states
624                        .map(|xm| xm[offset + (output_pos >> 5) as usize] >> (output_pos & 31) & 1 != 0)
625                        .unwrap_or(false);
626                    (v, x)
627                }
628            };
629
630            // Encode: 0=V0, 1=V1, 3=X
631            let encoded = if is_x { 3 } else { value_new };
632            if encoded == last_val[i] {
633                continue;
634            }
635            last_val[i] = encoded;
636            if is_x {
637                x_output_count += 1;
638            }
639            writer
640                .change_scalar(
641                    vid,
642                    match (is_x, value_new) {
643                        (true, _) => Value::X,
644                        (false, 1) => Value::V1,
645                        _ => Value::V0,
646                    },
647                )
648                .unwrap();
649        }
650    }
651    if x_output_count > 0 {
652        clilog::warn!(
653            "VCD output contains {} X-value transitions across all signals",
654            x_output_count
655        );
656    }
657}

707    #[test]
708    fn test_expand_split_roundtrip() {
709        // 2-word state, 1 input at bit 0, 1 DFF output at bit 32
710        let rio = 2;
711        let script = make_xprop_test_script(rio, &[0], &[32]);
712
713        // 3 snapshots (2 cycles + trailing), value states only
714        let value_states: Vec<u32> = vec![
715            0x0000_0001, 0x0000_0000, // snap 0: input bit 0 = 1
716            0x0000_0001, 0x0000_0001, // snap 1: input bit 0 = 1, dff bit 0 = 1
717            0x0000_0000, 0x0000_0001, // snap 2: input bit 0 = 0, dff bit 0 = 1
718        ];
719
720        let expanded = expand_states_for_xprop(&value_states, &script);
721        assert_eq!(expanded.len(), 3 * rio as usize * 2);
722
723        // Split back
724        let (values, xmasks) = split_xprop_states(&expanded, rio as usize);

726        assert_eq!(values, value_states, "values should match original");
727
728        // X-mask template: bit 0 = input → cleared (0), bit 32 = DFF → set (1)
729        // Word 0 of xmask: bit 0 cleared → 0xFFFF_FFFE
730        // Word 1 of xmask: all bits set → 0xFFFF_FFFF (DFF output at bit 0 of word 1)
731        for snap in 0..3 {

733            let xm_word1 = xmasks[snap * rio as usize + 1];
734            assert_eq!(
735                xm_word0, 0xFFFF_FFFE,
736                "snap {}: word 0 should have bit 0 cleared (input)",
737                snap
738            );
739            assert_eq!(
740                xm_word1, 0xFFFF_FFFF,
741                "snap {}: word 1 should be all-X (DFF output)",
742                snap
743            );

793        let rio = 4;

795        // DFF outputs at bits 32, 33, 96 (word 1 bits 0,1; word 3 bit 0)

799        let expanded = expand_states_for_xprop(&value_states, &script);
800

802
803        // Check xmask template (second half of expanded)
804        let xmask = &expanded[rio as usize..];
805
806        // Word 0: bits 0 and 5 are inputs → should be cleared
807        let expected_word0 = 0xFFFF_FFFF & !(1u32 << 0) & !(1u32 << 5);

813        // Word 1: no inputs → all 0xFFFFFFFF

816        // Word 2: bit 0 is input → cleared

818        assert_eq!(
819            xmask[2], expected_word2,
820            "word 2: input bit 0 should be cleared"
821        );
822
823        // Word 3: no inputs → all 0xFFFFFFFF
824        assert_eq!(xmask[3], 0xFFFF_FFFF, "word 3: no inputs, all X");
