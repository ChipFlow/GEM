17/// Work item for iterative DFS traversal.
18/// Using two-phase approach: Visit pushes dependencies, Process computes outputs.
19#[derive(Clone, Copy)]
20enum WorkItem {
21    /// First phase: check if visited, mark in-stack, push Process then dependencies
22    Visit(usize),
23    /// Second phase: all dependencies ready, compute output, clear in-stack
24    Process(usize),
25}

96/// A type of endpoint group. can be a primary output-related pin,
97/// a D flip-flop, a ram block, or a simulation control node.
98///
99/// A group means a task for the partition to complete.
100/// For primary output pins, the task is just to store.
101/// For DFFs, the task is to store only when the clock is enable.
102/// For RAMBlocks, the task is to simulate a sync SRAM.
103/// For SimControl, the task is to check the condition and fire an event.
104/// A StagedIOPin indicates a temporary live pin between different
105/// major stages but reside in the same simulated cycle.
106#[derive(Debug, Copy, Clone)]
107pub enum EndpointGroup<'i> {
108    PrimaryOutput(usize),
109    DFF(&'i DFF),
110    RAMBlock(&'i RAMBlock),
111    SimControl(&'i SimControlNode),
112    Display(&'i DisplayNode),
113    StagedIOPin(usize),
114}
115
116impl EndpointGroup<'_> {
117    /// Enumerate all related aigpin inputs for this endpoint group.
118    ///
119    /// The enumerated inputs may have duplicates.
120    pub fn for_each_input(self, mut f_nz: impl FnMut(usize)) {
121        let mut f = |i| {
122            if i >= 1 {
123                f_nz(i);
124            }
125        };
126        match self {
127            Self::PrimaryOutput(idx) => f(idx >> 1),
128            Self::DFF(dff) => {
129                f(dff.en_iv >> 1);
130                f(dff.d_iv >> 1);
131            }
132            Self::RAMBlock(ram) => {
133                f(ram.port_r_en_iv >> 1);
134                for i in 0..13 {
135                    f(ram.port_r_addr_iv[i] >> 1);
136                    f(ram.port_w_addr_iv[i] >> 1);
137                }
138                for i in 0..32 {
139                    f(ram.port_w_wr_en_iv[i] >> 1);
140                    f(ram.port_w_wr_data_iv[i] >> 1);
141                }
142            }
143            Self::SimControl(ctrl) => {
144                f(ctrl.condition_iv >> 1);
145            }
146            Self::Display(disp) => {
147                f(disp.enable_iv >> 1);
148                for &arg_iv in &disp.args_iv {
149                    f(arg_iv >> 1);
150                }
151            }
152            Self::StagedIOPin(idx) => f(idx),
153        }
154    }
155}

157/// The driver type of an AIG pin.
158#[derive(Debug, Clone)]
159pub enum DriverType {
160    /// Driven by an and gate.
161    ///
162    /// The inversion bit is stored as the last bits in
163    /// two input indices.
164    ///
165    /// Only this type has combinational fan-in.
166    AndGate(usize, usize),
167    /// Driven by a primary input port (with its netlistdb id).
168    InputPort(usize),
169    /// Driven by a clock flag (with clock port netlistdb id, and pos/negedge)
170    InputClockFlag(usize, u8),
171    /// Driven by a DFF (with its index)
172    DFF(usize),
173    /// Driven by a 13-bit by 32-bit RAM block (with its index)
174    SRAM(usize),
175    /// Tie0: tied to zero. Only the 0-th aig pin is allowed to have this.
176    Tie0,
177}

179/// An AIG associated with a netlistdb.
180#[derive(Debug)]
181pub struct AIG {
182    /// The number of AIG pins.
183    ///
184    /// This number might be smaller than num_pins in netlistdb,
185    /// because inverters and buffers are merged when possible.
186    /// It might also be larger because we may add mux circuits.
187    ///
188    /// AIG pins are numbered from 1 to num_aigpins inclusive.
189    /// The AIG pin id zero (0) is tied to 0.
190    ///
191    /// AIG pins are guaranteed to have topological order.
192    pub num_aigpins: usize,
193    /// The mapping from a netlistdb pin to an AIG pin.
194    ///
195    /// The inversion bit is stored as the last bit.
196    /// E.g., `pin2aigpin_iv[pin_id] = aigpin_id << 1 | invert`.
197    pub pin2aigpin_iv: Vec<usize>,
198    /// The clock pins map. Every clock pin has a pair of flag pins
199    /// showing if they are posedge/negedge.
200    ///
201    /// The flag pin can be empty which means the circuit is not
202    /// active with that edge.
203    pub clock_pin2aigpins: IndexMap<usize, (usize, usize)>,
204    /// The driver types of AIG pins.
205    pub drivers: Vec<DriverType>,
206    /// A cache for identical and gates.
207    pub and_gate_cache: IndexMap<(usize, usize), usize>,
208    /// Unique primary output aigpin indices
209    pub primary_outputs: IndexSet<usize>,
210    /// The D flip-flops (DFFs), indexed by cell id
211    pub dffs: IndexMap<usize, DFF>,
212    /// The SRAMs, indexed by cell id
213    pub srams: IndexMap<usize, RAMBlock>,
214    /// The simulation control nodes ($stop/$finish), indexed by cell id
215    pub simcontrols: IndexMap<usize, SimControlNode>,
216    /// The display nodes ($display/$write), indexed by cell id
217    pub displays: IndexMap<usize, DisplayNode>,
218    /// The fanout CSR start array.
219    pub fanouts_start: Vec<usize>,
220    /// The fanout CSR array.
221    pub fanouts: Vec<usize>,
222
223    // === Timing Analysis Fields ===
224    /// Arrival times for each AIG pin: (min_ps, max_ps).
225    /// Computed during STA traversal. Index 0 is unused (Tie0 has arrival 0).
226    /// Empty until `compute_timing()` is called.
227    pub arrival_times: Vec<(u64, u64)>,
228
229    /// Gate delays for each AIG pin: (rise_ps, fall_ps).
230    /// Loaded from Liberty library. Used during timing propagation.
231    pub gate_delays: Vec<(u64, u64)>,
232
233    /// Setup slack for each DFF (indexed by position in `dffs`).
234    /// Negative values indicate timing violations.
235    /// Computed as: clock_arrival - data_arrival - setup_time
236    pub setup_slacks: Vec<i64>,
237
238    /// Hold slack for each DFF (indexed by position in `dffs`).
239    /// Negative values indicate timing violations.
240    /// Computed as: data_arrival - clock_arrival - hold_time
241    pub hold_slacks: Vec<i64>,
242
243    /// DFF timing constraints loaded from Liberty.
244    pub dff_timing: Option<crate::liberty_parser::DFFTiming>,
245
246    /// Clock period in picoseconds (for STA calculations).
247    /// Default is 1000ps (1ns) if not specified.
248    pub clock_period_ps: u64,
249
250    // === SDF Back-Annotation Support ===
251    /// Maps AIG pin index â†’ list of (netlistdb cell_id, cell_type, output_pin_name).
252    /// Accumulated: inverters sharing an AIG pin (via invert-bit reuse) push
253    /// their origins rather than overwriting. During SDF loading, delays from
254    /// all origins are summed (they form a serial chain, not parallel paths).
255    /// Internal AND gates from multi-gate decompositions get empty Vec (zero delay).
256    /// Index 0 is Tie0 (empty). Populated during AIG construction from netlistdb.
257    pub aigpin_cell_origins: Vec<Vec<(usize, String, String)>>,
258}

287impl AIG {
288    fn add_aigpin(&mut self, driver: DriverType) -> usize {
289        self.num_aigpins += 1;
290        self.drivers.push(driver);
291        self.aigpin_cell_origins.push(Vec::new());
292        self.num_aigpins
293    }
294
295    fn add_and_gate(&mut self, a: usize, b: usize) -> usize {
296        assert_ne!(a | 1, usize::MAX);
297        assert_ne!(b | 1, usize::MAX);
298        if a == 0 || b == 0 {
299            return 0;
300        }
301        if a == 1 {
302            return b;
303        }
304        if b == 1 {
305            return a;
306        }
307        let (a, b) = if a < b { (a, b) } else { (b, a) };
308        if let Some(o) = self.and_gate_cache.get(&(a, b)) {
309            return o << 1;
310        }
311        let aigpin = self.add_aigpin(DriverType::AndGate(a, b));
312        self.and_gate_cache.insert((a, b), aigpin);
313        aigpin << 1
314    }

316    /// given a clock pin, trace back to clock root and return its
317    /// enable signal (with invert bit).
318    ///
319    /// if result is 0, that means the pin is dangled.
320    /// if an error occurs because of a undecipherable multi-input cell,
321    /// we will return in error the last output pin index of that cell.
322    fn trace_clock_pin(
323        &mut self,
324        netlistdb: &NetlistDB,
325        start_pinid: usize,
326        start_is_negedge: bool,
327        // should we ignore cklnqd in this tracing.
328        // if set to true, we will treat cklnqd as a simple buffer.
329        // otherwise, we assert that cklnqd/en is already built in
330        // our aig mapping (pin2aigpin_iv).
331        ignore_cklnqd: bool,
332    ) -> Result<usize, usize> {
333        // Iterative implementation using explicit stack
334        // Each entry: (pinid, is_negedge, is_processing_cklnqd, cklnqd_cp_result)
335        // For CKLNQD, we need to first trace CP, then combine with EN
336        let mut current_pinid = start_pinid;
337        let mut current_is_negedge = start_is_negedge;
338
339        // Stack for CKLNQD cells that need post-processing
340        // (pinid of EN pin, is_negedge at that point)
341        let mut cklnqd_stack: Vec<(usize, bool)> = Vec::new();
342
343        let mut iteration = 0;
344        let mut visited_pins: std::collections::HashSet<usize> = std::collections::HashSet::new();
345        let mut path: Vec<(usize, String)> = Vec::new();
346        loop {
347            iteration += 1;
348            {
349                let pin_name = netlistdb.pinnames[current_pinid].dbg_fmt_pin();
350                let dir = if netlistdb.pindirect[current_pinid] == Direction::I {
351                    "I"
352                } else {
353                    "O"
354                };
355                path.push((current_pinid, format!("{}:{}", pin_name, dir)));
356            }
357            if visited_pins.contains(&current_pinid) {
358                panic!(
359                    "trace_clock_pin cycle detected!\nstart={}, iteration={}\npath:\n{}",
360                    start_pinid,
361                    iteration,
362                    path.iter()
363                        .enumerate()
364                        .map(|(i, (pid, name))| format!("  {:3}: pin {} - {}", i, pid, name))
365                        .collect::<Vec<_>>()
366                        .join("\n")
367                );
368            }
369            visited_pins.insert(current_pinid);
370
371            if iteration > 100000 {
372                panic!(
373                    "trace_clock_pin infinite loop detected: pinid={}, start={}, iteration={}",
374                    current_pinid, start_pinid, iteration
375                );
376            }
377            let pinid = current_pinid;
378            let is_negedge = current_is_negedge;
379
380            if netlistdb.pindirect[pinid] == Direction::I {
381                let netid = netlistdb.pin2net[pinid];
382                if Some(netid) == netlistdb.net_zero || Some(netid) == netlistdb.net_one {
383                    // Reached a constant - process any pending CKLNQD
384                    let mut result = 0usize;
385                    while let Some((en_pin, _)) = cklnqd_stack.pop() {
386                        if ignore_cklnqd {
387                            // Just return the traced clock, ignore enable
388                        } else {
389                            let en_iv = self.pin2aigpin_iv[en_pin];
390                            assert_ne!(en_iv, usize::MAX, "clken not built");
391                            result = self.add_and_gate(result, en_iv);
392                        }
393                    }
394                    return Ok(result);
395                }
396                // Follow net to driving pin - find the output pin (driver) on the net
397                let net_pins_start = netlistdb.net2pin.start[netid];
398                let net_pins_end = if netid + 1 < netlistdb.net2pin.start.len() {
399                    netlistdb.net2pin.start[netid + 1]
400                } else {
401                    netlistdb.net2pin.items.len()
402                };
403                let mut driver_pin = None;
404                for &np in &netlistdb.net2pin.items[net_pins_start..net_pins_end] {
405                    // Check if this pin is an output (driver)
406                    if netlistdb.pindirect[np] == Direction::O {
407                        driver_pin = Some(np);
408                        break;
409                    }
410                    // Also check for primary input (cell 0)
411                    if netlistdb.pin2cell[np] == 0 {
412                        driver_pin = Some(np);
413                        break;
414                    }
415                }
416                if let Some(dp) = driver_pin {
417                    current_pinid = dp;
418                    continue;
419                }
420                // Net has no driver - treat this input pin as a primary input (clock source)
421                // This handles floating nets / undriven primary inputs
422                // Create a clock signal for this undriven net
423                let clkentry = self
424                    .clock_pin2aigpins
425                    .entry(pinid)
426                    .or_insert((usize::MAX, usize::MAX));
427                let clksignal = match is_negedge {
428                    false => clkentry.0,
429                    true => clkentry.1,
430                };
431                let mut result = if clksignal != usize::MAX {
432                    clksignal << 1
433                } else {
434                    let aigpin =
435                        self.add_aigpin(DriverType::InputClockFlag(pinid, is_negedge as u8));
436                    let clkentry = self.clock_pin2aigpins.get_mut(&pinid).unwrap();
437                    let clksignal = match is_negedge {
438                        false => &mut clkentry.0,
439                        true => &mut clkentry.1,
440                    };
441                    *clksignal = aigpin;
442                    aigpin << 1
443                };
444
445                // Process any pending CKLNQD
446                while let Some((en_pin, _)) = cklnqd_stack.pop() {
447                    if !ignore_cklnqd {
448                        let en_iv = self.pin2aigpin_iv[en_pin];
449                        assert_ne!(en_iv, usize::MAX, "clken not built");
450                        result = self.add_and_gate(result, en_iv);
451                    }
452                }
453                return Ok(result);
454            }
455
456            let cellid = netlistdb.pin2cell[pinid];
457            if cellid == 0 {
458                // Reached primary input - this is the clock source
459                let clkentry = self
460                    .clock_pin2aigpins
461                    .entry(pinid)
462                    .or_insert((usize::MAX, usize::MAX));
463                let clksignal = match is_negedge {
464                    false => clkentry.0,
465                    true => clkentry.1,
466                };
467                let mut result = if clksignal != usize::MAX {
468                    clksignal << 1
469                } else {
470                    let aigpin =
471                        self.add_aigpin(DriverType::InputClockFlag(pinid, is_negedge as u8));
472                    let clkentry = self.clock_pin2aigpins.get_mut(&pinid).unwrap();
473                    let clksignal = match is_negedge {
474                        false => &mut clkentry.0,
475                        true => &mut clkentry.1,
476                    };
477                    *clksignal = aigpin;
478                    aigpin << 1
479                };
480
481                // Process any pending CKLNQD
482                while let Some((en_pin, _)) = cklnqd_stack.pop() {
483                    if !ignore_cklnqd {
484                        let en_iv = self.pin2aigpin_iv[en_pin];
485                        assert_ne!(en_iv, usize::MAX, "clken not built");
486                        result = self.add_and_gate(result, en_iv);
487                    }
488                }
489                return Ok(result);
490            }
491
492            let mut pin_a = usize::MAX;
493            let mut pin_cp = usize::MAX;
494            let mut pin_en = usize::MAX;
495            let celltype = netlistdb.celltypes[cellid].as_str();
496
497            // Determine if this is a clock buffer/inverter (AIGPDK or SKY130)
498            let is_inv = celltype == "INV"
499                || (is_sky130_cell(celltype) && {
500                    let ct = extract_cell_type(celltype);
501                    ct.starts_with("inv") || ct.starts_with("clkinv")
502                });
503            let is_buf = celltype == "BUF"
504                || (is_sky130_cell(celltype) && {
505                    let ct = extract_cell_type(celltype);
506                    ct.starts_with("buf")
507                        || ct.starts_with("clkbuf")
508                        || ct.starts_with("clkdlybuf")
509                        || ct.starts_with("lpflow_isobufsrc")
510                        || ct.starts_with("lpflow_inputiso")
511                });
512
513            if !is_inv && !is_buf && celltype != "CKLNQD" {
514                clilog::error!(
515                    "cell type {} not supported on clock path. expecting only INV, BUF, or CKLNQD (or SKY130 equivalents)",
516                    celltype
517                );
518                return Err(pinid);
519            }
520
521            for ipin in netlistdb.cell2pin.iter_set(cellid) {
522                if netlistdb.pindirect[ipin] == Direction::I {
523                    match netlistdb.pinnames[ipin].1.as_str() {
524                        "A" => pin_a = ipin,
525                        "CP" => pin_cp = ipin,
526                        "E" => pin_en = ipin,
527                        i @ _ => {
528                            clilog::error!(
529                                "input pin {} unexpected for ck element {}",
530                                i,
531                                celltype
532                            );
533                            return Err(ipin);
534                        }
535                    }
536                }
537            }
538
539            if is_inv {
540                assert_ne!(pin_a, usize::MAX);
541                current_pinid = pin_a;
542                current_is_negedge = !is_negedge;
543            } else if is_buf {
544                assert_ne!(pin_a, usize::MAX);
545                current_pinid = pin_a;
546                // is_negedge stays the same
547            } else if celltype == "CKLNQD" {
548                assert_ne!(pin_cp, usize::MAX);
549                assert_ne!(pin_en, usize::MAX);
550                // Push CKLNQD for post-processing, continue with CP
551                cklnqd_stack.push((pin_en, is_negedge));
552                current_pinid = pin_cp;
553                // is_negedge stays the same
554            } else {
555                unreachable!()
556            }
557        }
558    }

560    /// Iteratively add AIG pins for netlistdb pins using explicit stack.
561    ///
562    /// for sequential logics like DFF and RAM,
563    /// 1. their netlist pin inputs are not patched,
564    /// 2. their aig pin inputs (in dffs and srams arrays) will be
565    ///    patched to include mux -- but not inside this function.
566    /// 3. their netlist/aig outputs are directly built here,
567    ///    with possible patches for asynchronous DFFSR polyfill.
568    fn dfs_netlistdb_build_aig(
569        &mut self,
570        netlistdb: &NetlistDB,
571        topo_vis: &mut Vec<bool>,
572        topo_instack: &mut Vec<bool>,
573        start_pinid: usize,
574        pdk_models: Option<&PdkModels>,
575    ) {
576        let mut work_stack: Vec<WorkItem> = vec![WorkItem::Visit(start_pinid)];
577
578        while let Some(item) = work_stack.pop() {
579            match item {
580                WorkItem::Visit(pinid) => {
581                    if topo_vis[pinid] {
582                        continue;
583                    }
584                    if topo_instack[pinid] {
585                        panic!(
586                            "circuit has a loop around pin {}",
587                            netlistdb.pinnames[pinid].dbg_fmt_pin()
588                        );
589                    }
590
591                    topo_vis[pinid] = true;
592                    topo_instack[pinid] = true;
593
594                    let netid = netlistdb.pin2net[pinid];
595                    let cellid = netlistdb.pin2cell[pinid];
596                    let celltype = netlistdb.celltypes[cellid].as_str();
597
598                    // Handle input pins
599                    if netlistdb.pindirect[pinid] == Direction::I {
600                        if Some(netid) == netlistdb.net_zero {
601                            self.pin2aigpin_iv[pinid] = 0;
602                            topo_instack[pinid] = false;
603                        } else if Some(netid) == netlistdb.net_one {
604                            self.pin2aigpin_iv[pinid] = 1;
605                            topo_instack[pinid] = false;
606                        } else {
607                            // Find the actual driver pin on the net
608                            let net_pins_start = netlistdb.net2pin.start[netid];
609                            let net_pins_end = if netid + 1 < netlistdb.net2pin.start.len() {
610                                netlistdb.net2pin.start[netid + 1]
611                            } else {
612                                netlistdb.net2pin.items.len()
613                            };
614                            let mut driver_pin = None;
615                            for &np in &netlistdb.net2pin.items[net_pins_start..net_pins_end] {
616                                // Check if this pin is an output (driver)
617                                if netlistdb.pindirect[np] == Direction::O {
618                                    driver_pin = Some(np);
619                                    break;
620                                }
621                                // Also check for primary input (cell 0)
622                                if netlistdb.pin2cell[np] == 0 {
623                                    driver_pin = Some(np);
624                                    break;
625                                }
626                            }
627                            if let Some(root) = driver_pin {
628                                // Push process, then dependency
629                                work_stack.push(WorkItem::Process(pinid));
630                                work_stack.push(WorkItem::Visit(root));
631                            } else {
632                                // Net has no driver - tie to 0
633                                // This handles floating nets / undriven signals in post-P&R netlists
634                                self.pin2aigpin_iv[pinid] = 0;
635                                topo_instack[pinid] = false;
636                            }
637                        }
638                        continue;
639                    }
640
641                    // Handle primary input ports (cellid == 0, output direction)
642                    if cellid == 0 {
643                        let aigpin = self.add_aigpin(DriverType::InputPort(pinid));
644                        self.pin2aigpin_iv[pinid] = aigpin << 1;
645                        topo_instack[pinid] = false;
646                        continue;
647                    }
648
649                    // Handle AIGPDK DFF/DFFSR
650                    if matches!(celltype, "DFF" | "DFFSR") {
651                        // Pre-create DFF Q output
652                        let q = self.add_aigpin(DriverType::DFF(cellid));
653                        // Record cell origin for DFF Q output (for SDF CLKâ†’Q delay)
654                        self.aigpin_cell_origins[q].push((
655                            cellid,
656                            celltype.to_string(),
657                            "Q".to_string(),
658                        ));
659                        let dff = self.dffs.entry(cellid).or_default();
660                        dff.q = q;
661
662                        // Push process then S/R dependencies
663                        work_stack.push(WorkItem::Process(pinid));
664                        for dep_pinid in netlistdb.cell2pin.iter_set(cellid) {
665                            if matches!(netlistdb.pinnames[dep_pinid].1.as_str(), "S" | "R") {
666                                work_stack.push(WorkItem::Visit(dep_pinid));
667                            }
668                        }
669                        continue;
670                    }
671
672                    // Handle LATCH
673                    if celltype == "LATCH" {
674                        panic!(
675                            "latches are intentionally UNSUPPORTED by GEM, \
676                                except in identified gated clocks. \n\
677                                you can link a FF&MUX-based LATCH module, \
678                                but most likely that is NOT the right solution. \n\
679                                check all your assignments inside always@(*) block \
680                                to make sure they cover all scenarios."
681                        );
682                    }
683
684                    // Handle SRAM
685                    if celltype == "$__RAMGEM_SYNC_" {
686                        let o = self.add_aigpin(DriverType::SRAM(cellid));
687                        self.pin2aigpin_iv[pinid] = o << 1;
688                        assert_eq!(netlistdb.pinnames[pinid].1.as_str(), "PORT_R_RD_DATA");
689                        // Record cell origin for SRAM data output
690                        self.aigpin_cell_origins[o].push((
691                            cellid,
692                            celltype.to_string(),
693                            "PORT_R_RD_DATA".to_string(),
694                        ));
695                        let sram = self.srams.entry(cellid).or_default();
696                        sram.port_r_rd_data[netlistdb.pinnames[pinid].2.unwrap() as usize] = o;
697                        topo_instack[pinid] = false;
698                        continue;
699                    }
700
701                    // Handle CKLNQD
702                    if celltype == "CKLNQD" {
703                        let mut prev_cp = usize::MAX;
704                        let mut prev_en = usize::MAX;
705                        for dep_pinid in netlistdb.cell2pin.iter_set(cellid) {
706                            match netlistdb.pinnames[dep_pinid].1.as_str() {
707                                "CP" => prev_cp = dep_pinid,
708                                "E" => prev_en = dep_pinid,
709                                _ => {}
710                            }
711                        }
712                        assert_ne!(prev_cp, usize::MAX);
713                        assert_ne!(prev_en, usize::MAX);
714                        // Push process then dependencies
715                        work_stack.push(WorkItem::Process(pinid));
716                        work_stack.push(WorkItem::Visit(prev_cp));
717                        work_stack.push(WorkItem::Visit(prev_en));
718                        continue;
719                    }
720
721                    // Handle GEM_ASSERT / GEM_DISPLAY
722                    if celltype == "GEM_ASSERT" || celltype == "GEM_DISPLAY" {
723                        // Push process then all input pin dependencies
724                        work_stack.push(WorkItem::Process(pinid));
725                        for dep_pinid in netlistdb.cell2pin.iter_set(cellid) {
726                            if netlistdb.pindirect[dep_pinid] == Direction::I {
727                                work_stack.push(WorkItem::Visit(dep_pinid));
728                            }
729                        }
730                        continue;
731                    }
732
733                    // Handle SKY130 cells
734                    if is_sky130_cell(celltype) {
735                        // Get dependencies based on cell type
736                        let deps = self.get_sky130_dependencies(netlistdb, pinid, cellid, celltype);
737                        // Do any pre-processing
738                        self.sky130_preprocess(netlistdb, pinid, cellid, celltype);
739                        // Push process then dependencies
740                        work_stack.push(WorkItem::Process(pinid));
741                        for dep in deps {
742                            work_stack.push(WorkItem::Visit(dep));
743                        }
744                        continue;
745                    }
746
747                    // Handle AIGPDK combinational cells (AND2, INV, BUF)
748                    let mut prev_a = usize::MAX;
749                    let mut prev_b = usize::MAX;
750                    for dep_pinid in netlistdb.cell2pin.iter_set(cellid) {
751                        match netlistdb.pinnames[dep_pinid].1.as_str() {
752                            "A" => prev_a = dep_pinid,
753                            "B" => prev_b = dep_pinid,
754                            _ => {}
755                        }
756                    }
757                    // Push process then dependencies
758                    work_stack.push(WorkItem::Process(pinid));
759                    if prev_b != usize::MAX {
760                        work_stack.push(WorkItem::Visit(prev_b));
761                    }
762                    if prev_a != usize::MAX {
763                        work_stack.push(WorkItem::Visit(prev_a));
764                    }
765                }
766
767                WorkItem::Process(pinid) => {
768                    let netid = netlistdb.pin2net[pinid];
769                    let cellid = netlistdb.pin2cell[pinid];
770                    let celltype = netlistdb.celltypes[cellid].as_str();
771
772                    // Process input pins (copy from driver)
773                    if netlistdb.pindirect[pinid] == Direction::I {
774                        let root = netlistdb.net2pin.items[netlistdb.net2pin.start[netid]];
775                        self.pin2aigpin_iv[pinid] = self.pin2aigpin_iv[root];
776                        if cellid == 0 {
777                            self.primary_outputs.insert(self.pin2aigpin_iv[pinid]);
778                        }
779                        topo_instack[pinid] = false;
780                        continue;
781                    }
782
783                    // Process AIGPDK DFF/DFFSR
784                    if matches!(celltype, "DFF" | "DFFSR") {
785                        let dff = self.dffs.get(&cellid).unwrap();
786                        let q = dff.q;
787                        let mut ap_s_iv = 1;
788                        let mut ap_r_iv = 1;
789                        let mut q_out = q << 1;
790
791                        for dep_pinid in netlistdb.cell2pin.iter_set(cellid) {
792                            let prev = self.pin2aigpin_iv[dep_pinid];
793                            match netlistdb.pinnames[dep_pinid].1.as_str() {
794                                "S" => ap_s_iv = prev,
795                                "R" => ap_r_iv = prev,
796                                _ => {}
797                            }
798                        }
799                        // AIGPDK DFFSR .lib says: clear="(!R)", preset="(!S)"
800                        // i.e. R=0 â†’ Q=0 (clear), S=0 â†’ Q=1 (preset)
801                        // Same active-low semantics as SKY130 RESET_B/SET_B.
802                        //   Q = AND(OR(Q_state, !S), R)
803                        q_out = self.add_and_gate(q_out ^ 1, ap_s_iv) ^ 1;
804                        q_out = self.add_and_gate(q_out, ap_r_iv);
805                        self.pin2aigpin_iv[pinid] = q_out;
806                        topo_instack[pinid] = false;
807                        continue;
808                    }
809
810                    // Process CKLNQD (no output pin defined)
811                    if celltype == "CKLNQD" {
812                        // do not define pin2aigpin_iv[pinid] which is CKLNQD/Q and unused in logic.
813                        topo_instack[pinid] = false;
814                        continue;
815                    }
816
817                    // Process GEM_ASSERT / GEM_DISPLAY (no output pin defined)
818                    if celltype == "GEM_ASSERT" || celltype == "GEM_DISPLAY" {
819                        topo_instack[pinid] = false;
820                        continue;
821                    }
822
823                    // Process SKY130 cells
824                    if is_sky130_cell(celltype) {
825                        self.sky130_postprocess(netlistdb, pinid, cellid, celltype, pdk_models);
826                        topo_instack[pinid] = false;
827                        continue;
828                    }
829
830                    // Process AIGPDK combinational cells
831                    let mut prev_a = usize::MAX;
832                    let mut prev_b = usize::MAX;
833                    for dep_pinid in netlistdb.cell2pin.iter_set(cellid) {
834                        match netlistdb.pinnames[dep_pinid].1.as_str() {
835                            "A" => prev_a = dep_pinid,
836                            "B" => prev_b = dep_pinid,
837                            _ => {}
838                        }
839                    }
840
841                    match celltype {
842                        "AND2_00_0" | "AND2_01_0" | "AND2_10_0" | "AND2_11_0" | "AND2_11_1" => {
843                            assert_ne!(prev_a, usize::MAX);
844                            assert_ne!(prev_b, usize::MAX);
845                            let name = netlistdb.celltypes[cellid].as_bytes();
846                            let iv_a = name[5] - b'0';
847                            let iv_b = name[6] - b'0';
848                            let iv_y = name[8] - b'0';
849                            let apid = self.add_and_gate(
850                                self.pin2aigpin_iv[prev_a] ^ (iv_a as usize),
851                                self.pin2aigpin_iv[prev_b] ^ (iv_b as usize),
852                            ) ^ (iv_y as usize);
853                            self.pin2aigpin_iv[pinid] = apid;
854                        }
855                        "INV" => {
856                            assert_ne!(prev_a, usize::MAX);
857                            self.pin2aigpin_iv[pinid] = self.pin2aigpin_iv[prev_a] ^ 1;
858                        }
859                        "BUF" => {
860                            assert_ne!(prev_a, usize::MAX);
861                            self.pin2aigpin_iv[pinid] = self.pin2aigpin_iv[prev_a];
862                        }
863                        _ => panic!("Unknown AIGPDK cell type: {}", celltype),
864                    }
865                    topo_instack[pinid] = false;
866                }
867            }
868        }
869    }

871    /// Get dependencies for a SKY130 cell that need to be visited before processing.
872    /// Uses SmallVec to avoid heap allocation for most cells (â‰¤6 inputs).
873    fn get_sky130_dependencies(
874        &self,
875        netlistdb: &NetlistDB,
876        pinid: usize,
877        cellid: usize,
878        celltype: &str,
879    ) -> SmallVec<[usize; 6]> {
880        let cell_type = extract_cell_type(celltype);
881        let output_pin_name = netlistdb.pinnames[pinid].1.as_str();
882
883        // Tie cells have no dependencies
884        if is_tie_cell(cell_type) {
885            return SmallVec::new();
886        }
887
888        // SRAM has no dependencies for output creation
889        if celltype.starts_with("CF_SRAM_") {
890            return SmallVec::new();
891        }
892
893        // Sequential cells: depend on SET_B/RESET_B pins
894        if is_sequential_cell(cell_type) {
895            let mut deps = SmallVec::new();
896            for dep_pinid in netlistdb.cell2pin.iter_set(cellid) {
897                let pin_name = netlistdb.pinnames[dep_pinid].1.as_str();
898                if matches!(pin_name, "SET_B" | "RESET_B") {
899                    deps.push(dep_pinid);
900                }
901            }
902            return deps;
903        }
904
905        // Multi-output cells (ha, fa, dfbbp)
906        if is_multi_output_cell(cell_type) {
907            // dfbbp Q_N depends on Q being built first
908            if cell_type == "dfbbp" && output_pin_name == "Q_N" {
909                // Find Q pin
910                for opin in netlistdb.cell2pin.iter_set(cellid) {
911                    if netlistdb.pinnames[opin].1.as_str() == "Q" {
912                        if self.pin2aigpin_iv[opin] == usize::MAX {
913                            let mut deps = SmallVec::new();
914                            deps.push(opin);
915                            return deps;
916                        }
917                        return SmallVec::new();
918                    }
919                }
920            }
921            // For other multi-output cells, depend on all input pins
922            let mut deps = SmallVec::new();
923            for dep_pinid in netlistdb.cell2pin.iter_set(cellid) {
924                if netlistdb.pindirect[dep_pinid] == Direction::I {
925                    deps.push(dep_pinid);
926                }
927            }
928            return deps;
929        }
930
931        // Combinational cells: depend on all input pins
932        let mut deps = SmallVec::new();
933        for dep_pinid in netlistdb.cell2pin.iter_set(cellid) {
934            if netlistdb.pindirect[dep_pinid] == Direction::I {
935                deps.push(dep_pinid);
936            }
937        }
938        deps
939    }

941    /// Pre-process for SKY130 cells (create output AIG pins before dependencies are visited).
942    fn sky130_preprocess(
943        &mut self,
944        netlistdb: &NetlistDB,
945        pinid: usize,
946        cellid: usize,
947        celltype: &str,
948    ) {
949        let cell_type = extract_cell_type(celltype);
950        let output_pin_name = netlistdb.pinnames[pinid].1.as_str();
951
952        // Tie cells - set output immediately
953        if is_tie_cell(cell_type) {
954            match output_pin_name {
955                "HI" => self.pin2aigpin_iv[pinid] = 1,
956                "LO" => self.pin2aigpin_iv[pinid] = 0,
957                _ => panic!("Unknown tie cell output: {}", output_pin_name),
958            }
959            return;
960        }
961
962        // SRAM output creation
963        if celltype.starts_with("CF_SRAM_") {
964            let o = self.add_aigpin(DriverType::SRAM(cellid));
965            self.pin2aigpin_iv[pinid] = o << 1;
966            assert_eq!(output_pin_name, "DO", "Expected DO output pin for CF_SRAM");
967            // Record cell origin for SRAM data output
968            self.aigpin_cell_origins[o].push((
969                cellid,
970                celltype.to_string(),
971                output_pin_name.to_string(),
972            ));
973            let sram = self.srams.entry(cellid).or_default();
974            let bit_idx = netlistdb.pinnames[pinid]
975                .2
976                .expect("DO pin must have bit index") as usize;
977            sram.port_r_rd_data[bit_idx] = o;
978            return;
979        }
980
981        // Sequential cells: pre-create DFF Q output
982        if is_sequential_cell(cell_type) {
983            let q = self.add_aigpin(DriverType::DFF(cellid));
984            // Record cell origin for DFF Q output (for SDF CLKâ†’Q delay)
985            self.aigpin_cell_origins[q].push((cellid, cell_type.to_string(), "Q".to_string()));
986            let dff = self.dffs.entry(cellid).or_default();
987            dff.q = q;
988            return;
989        }
990
991        // dfbbp with Q_N output: no pre-processing needed
992        // Other multi-output and combinational cells: no pre-processing needed
993    }

995    /// Post-process for SKY130 cells (compute output after dependencies are visited).
996    fn sky130_postprocess(
997        &mut self,
998        netlistdb: &NetlistDB,
999        pinid: usize,
1000        cellid: usize,
1001        celltype: &str,
1002        pdk_models: Option<&PdkModels>,
1003    ) {
1004        let cell_type = extract_cell_type(celltype);
1005
1006        // Tie cells and SRAM were already handled in preprocess
1007        if is_tie_cell(cell_type) || celltype.starts_with("CF_SRAM_") {
1008            return;
1009        }
1010
1011        // Sequential cells: apply reset/set logic to Q
1012        if is_sequential_cell(cell_type) {
1013            let dff = self.dffs.get(&cellid).unwrap();
1014            let q = dff.q;
1015            let mut ap_s_iv = 1;
1016            let mut ap_r_iv = 1;
1017            let mut q_out = q << 1;
1018
1019            for dep_pinid in netlistdb.cell2pin.iter_set(cellid) {
1020                let pin_name = netlistdb.pinnames[dep_pinid].1.as_str();
1021                let prev = self.pin2aigpin_iv[dep_pinid];
1022                match pin_name {
1023                    // SKY130 SET_B and RESET_B are active-low, same convention
1024                    // as AIGPDK's S and R pins. No inversion needed - the AIG
1025                    // formula OR(Q, NOT pin) / AND(result, pin) already handles
1026                    // active-low correctly:
1027                    //   RESET_B=0 (reset): AND(Q, 0) = 0  âœ“
1028                    //   RESET_B=1 (normal): AND(Q, 1) = Q  âœ“
1029                    "SET_B" => ap_s_iv = prev,
1030                    "RESET_B" => ap_r_iv = prev,
1031                    _ => {}
1032                }
1033            }
1034            q_out = self.add_and_gate(q_out ^ 1, ap_s_iv) ^ 1;
1035            q_out = self.add_and_gate(q_out, ap_r_iv);
1036            self.pin2aigpin_iv[pinid] = q_out;
1037            return;
1038        }
1039
1040        // Multi-output cells (ha, fa, dfbbp)
1041        if is_multi_output_cell(cell_type) {
1042            self.build_sky130_multi_output_postprocess(
1043                netlistdb, pinid, cellid, cell_type, pdk_models,
1044            );
1045            return;
1046        }
1047
1048        // Combinational cells: decompose and build
1049        // Use CellInputs struct instead of HashMap to avoid allocation
1050        let mut inputs = CellInputs::new();
1051        let mut input_count = 0;
1052        for ipin in netlistdb.cell2pin.iter_set(cellid) {
1053            // Include both explicit inputs (Direction::I) and unknown direction pins (Direction::U)
1054            // Unknown direction pins are treated as inputs for decomposition
1055            // Only skip explicit outputs (Direction::O)
1056            if netlistdb.pindirect[ipin] != Direction::O {
1057                let pin_name = netlistdb.pinnames[ipin].1.as_str();
1058                // Skip output pins by name (Y, X, Q, etc.)
1059                if !matches!(pin_name, "Y" | "X" | "Q" | "Q_N" | "SUM" | "COUT") {
1060                    inputs.set_pin(pin_name, self.pin2aigpin_iv[ipin]);
1061                    input_count += 1;
1062                }
1063            }
1064        }
1065
1066        // Debug: check if we got enough inputs
1067        if cell_type == "nand2" && inputs.a == usize::MAX {
1068            use netlistdb::GeneralHierName;
1069            let cell_name = netlistdb.cellnames[cellid].dbg_fmt_hier();
1070            let mut pins_info: Vec<String> = Vec::new();
1071            for ipin in netlistdb.cell2pin.iter_set(cellid) {
1072                let pin_name = netlistdb.pinnames[ipin].1.as_str();
1073                let dir = format!("{:?}", netlistdb.pindirect[ipin]);
1074                pins_info.push(format!("{}:{}:{}", pin_name, dir, ipin));
1075            }
1076            clilog::warn!(
1077                "nand2 cell {} missing A input. Pins: {:?}",
1078                cell_name,
1079                pins_info
1080            );
1081        }
1082        if input_count == 0 {
1083            use netlistdb::GeneralHierName;
1084            let cell_name = netlistdb.cellnames[cellid].dbg_fmt_hier();
1085            let mut pins_info: Vec<String> = Vec::new();
1086            for ipin in netlistdb.cell2pin.iter_set(cellid) {
1087                let pin_name = netlistdb.pinnames[ipin].1.as_str();
1088                let dir = format!("{:?}", netlistdb.pindirect[ipin]);
1089                pins_info.push(format!("{}:{}", pin_name, dir));
1090            }
1091            panic!(
1092                "No input pins found for cell {} ({}), type='{}'. Pins: {:?}",
1093                cellid, cell_name, cell_type, pins_info
1094            );
1095        }
1096
1097        let output_pin_name = netlistdb.pinnames[pinid].1.as_str();
1098        let pdk = pdk_models.expect(
1099            "PDK models required for SKY130 cell decomposition. \
1100             Ensure sky130_fd_sc_hd submodule is initialized.",
1101        );
1102        let decomp = decompose_with_pdk(cell_type, &inputs, output_pin_name, pdk);
1103
1104        // Use SmallVec for gate outputs - most cells have â‰¤5 gates
1105        let mut gate_outputs: SmallVec<[usize; 5]> = SmallVec::new();
1106        for (i, (a_ref, b_ref)) in decomp.and_gates.iter().enumerate() {
1107            let a_iv = match std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
1108                self.resolve_decomp_ref(*a_ref, &gate_outputs)
1109            })) {
1110                Ok(v) => v,
1111                Err(_) => {
1112                    use netlistdb::GeneralHierName;
1113                    let cell_name = netlistdb.cellnames[cellid].dbg_fmt_hier();
1114                    panic!(
1115                        "Failed resolve_decomp_ref for cell_type='{}', cell={}, gate {}, a_ref={}, gates_so_far={}, inputs={:?}",
1116                        cell_type, cell_name, i, a_ref, gate_outputs.len(), inputs
1117                    )
1118                }
1119            };
1120            let b_iv = match std::panic::catch_unwind(std::panic::AssertUnwindSafe(|| {
1121                self.resolve_decomp_ref(*b_ref, &gate_outputs)
1122            })) {
1123                Ok(v) => v,
1124                Err(_) => {
1125                    use netlistdb::GeneralHierName;
1126                    let cell_name = netlistdb.cellnames[cellid].dbg_fmt_hier();
1127                    panic!(
1128                        "Failed resolve_decomp_ref for cell_type='{}', cell={}, gate {}, b_ref={}, gates_so_far={}, inputs={:?}",
1129                        cell_type, cell_name, i, b_ref, gate_outputs.len(), inputs
1130                    )
1131                }
1132            };
1133            let gate_out = self.add_and_gate(a_iv, b_iv);
1134            gate_outputs.push(gate_out >> 1);
1135        }
1136
1137        let output_iv = if decomp.output_idx < 0 {
1138            // Negative index = reference to a gate we just created
1139            let gate_idx = (-decomp.output_idx - 1) as usize;
1140            gate_outputs[gate_idx] << 1
1141        } else {
1142            // Positive index = passthrough to an existing AIG pin
1143            // Need to encode as (index << 1) since output_idx is a raw index
1144            (decomp.output_idx as usize) << 1
1145        };
1146
1147        let final_iv = if decomp.output_inverted {
1148            output_iv ^ 1
1149        } else {
1150            output_iv
1151        };
1152
1153        self.pin2aigpin_iv[pinid] = final_iv;
1154
1155        // Record cell origin for SDF back-annotation.
1156        // The full IOPATH delay goes on the output AIG pin only;
1157        // internal AND gates from multi-gate decompositions keep None (zero delay).
1158        let output_aigpin = final_iv >> 1;
1159        if output_aigpin > 0 && output_aigpin < self.aigpin_cell_origins.len() {
1160            let output_pin_name = netlistdb.pinnames[pinid].1.as_str();
1161            self.aigpin_cell_origins[output_aigpin].push((
1162                cellid,
1163                cell_type.to_string(),
1164                output_pin_name.to_string(),
1165            ));
1166        }
1167    }

1169    /// Post-process for SKY130 multi-output cells (ha, fa, dfbbp).
1170    fn build_sky130_multi_output_postprocess(
1171        &mut self,
1172        netlistdb: &NetlistDB,
1173        output_pinid: usize,
1174        cellid: usize,
1175        cell_type: &str,
1176        pdk_models: Option<&PdkModels>,
1177    ) {
1178        let output_pin_name = netlistdb.pinnames[output_pinid].1.as_str();
1179
1180        // dfbbp is sequential - handle inline (not in PDK combinational models)
1181        if cell_type == "dfbbp" {
1182            if output_pin_name == "Q_N" {
1183                // Q_N is inverted Q
1184                for opin in netlistdb.cell2pin.iter_set(cellid) {
1185                    if netlistdb.pinnames[opin].1.as_str() == "Q" {
1186                        self.pin2aigpin_iv[output_pinid] = self.pin2aigpin_iv[opin] ^ 1;
1187                        return;
1188                    }
1189                }
1190                panic!("dfbbp missing Q pin");
1191            } else {
1192                // Q output - build DFF logic
1193                let dff = self.dffs.get(&cellid).unwrap();
1194                let q = dff.q;
1195                let mut ap_s_iv = 1;
1196                let mut ap_r_iv = 1;
1197                let mut q_out = q << 1;
1198
1199                for dep_pinid in netlistdb.cell2pin.iter_set(cellid) {
1200                    let pin_name = netlistdb.pinnames[dep_pinid].1.as_str();
1201                    let prev = self.pin2aigpin_iv[dep_pinid];
1202                    match pin_name {
1203                        "SET_B" => ap_s_iv = prev,
1204                        "RESET_B" => ap_r_iv = prev,
1205                        _ => {}
1206                    }
1207                }
1208                q_out = self.add_and_gate(q_out ^ 1, ap_s_iv) ^ 1;
1209                q_out = self.add_and_gate(q_out, ap_r_iv);
1210                self.pin2aigpin_iv[output_pinid] = q_out;
1211            }
1212            return;
1213        }
1214
1215        // Use PDK models for ha/fa
1216        let pdk = pdk_models.expect(
1217            "PDK models required for SKY130 multi-output cell decomposition. \
1218             Ensure sky130_fd_sc_hd submodule is initialized.",
1219        );
1220        let mut inputs = CellInputs::new();
1221        for ipin in netlistdb.cell2pin.iter_set(cellid) {
1222            if netlistdb.pindirect[ipin] == Direction::I {
1223                let pin_name = netlistdb.pinnames[ipin].1.as_str();
1224                inputs.set_pin(pin_name, self.pin2aigpin_iv[ipin]);
1225            }
1226        }
1227        let decomp = decompose_with_pdk(cell_type, &inputs, output_pin_name, pdk);
1228        // Build the AIG gates from the decomposition
1229        let mut gate_outputs: SmallVec<[usize; 5]> = SmallVec::new();
1230        for (a_ref, b_ref) in &decomp.and_gates {
1231            let a_iv = self.resolve_decomp_ref(*a_ref, &gate_outputs);
1232            let b_iv = self.resolve_decomp_ref(*b_ref, &gate_outputs);
1233            let gate_out = self.add_and_gate(a_iv, b_iv);
1234            gate_outputs.push(gate_out >> 1);
1235        }
1236        let output_iv = if decomp.output_idx < 0 {
1237            let gate_idx = (-decomp.output_idx - 1) as usize;
1238            gate_outputs[gate_idx] << 1
1239        } else {
1240            (decomp.output_idx as usize) << 1
1241        };
1242        let final_iv = if decomp.output_inverted {
1243            output_iv ^ 1
1244        } else {
1245            output_iv
1246        };
1247        self.pin2aigpin_iv[output_pinid] = final_iv;
1248
1249        // Record cell origin for SDF back-annotation
1250        let output_aigpin = final_iv >> 1;
1251        if output_aigpin > 0 && output_aigpin < self.aigpin_cell_origins.len() {
1252            self.aigpin_cell_origins[output_aigpin].push((
1253                cellid,
1254                cell_type.to_string(),
1255                output_pin_name.to_string(),
1256            ));
1257        }
1258    }

1260    /// Resolve a decomposition reference to an aigpin_iv value.
1261    fn resolve_decomp_ref(&self, ref_val: i64, gate_outputs: &[usize]) -> usize {
1262        if ref_val < 0 {
1263            // Reference to intermediate gate output
1264            // The inversion bit might be encoded in ref_val
1265            let base_ref = ref_val | 1; // Clear potential inversion bit for index calc
1266            let gate_idx = (-(base_ref >> 1) - 1) as usize;
1267            if gate_idx >= gate_outputs.len() {
1268                panic!(
1269                    "resolve_decomp_ref: ref_val={}, base_ref={}, gate_idx={}, gate_outputs.len()={}",
1270                    ref_val, base_ref, gate_idx, gate_outputs.len()
1271                );
1272            }
1273            let invert = (ref_val & 1) != (base_ref & 1);
1274            let gate_out = gate_outputs[gate_idx] << 1;
1275            if invert {
1276                gate_out ^ 1
1277            } else {
1278                gate_out
1279            }
1280        } else {
1281            // Direct input reference (already has inversion bit)
1282            ref_val as usize
1283        }
1284    }

1327    fn from_netlistdb_impl(netlistdb: &NetlistDB, pdk_models: Option<&PdkModels>) -> AIG {
1328        let mut aig = AIG {
1329            num_aigpins: 0,
1330            pin2aigpin_iv: vec![usize::MAX; netlistdb.num_pins],
1331            drivers: vec![DriverType::Tie0],
1332            aigpin_cell_origins: vec![Vec::new()], // Tie0 has no cell origin
1333            ..Default::default()
1334        };
1335
1336        clilog::info!(
1337            "Starting clock tracing for {} cells...",
1338            netlistdb.num_cells
1339        );
1340        let clock_start = std::time::Instant::now();
1341        let mut seq_count = 0;
1342        let mut trace_count = 0;
1343
1344        for cellid in 1..netlistdb.num_cells {
1345            let celltype = netlistdb.celltypes[cellid].as_str();
1346
1347            // Check if this is a sequential element (AIGPDK or SKY130)
1348            let is_aigpdk_seq = matches!(celltype, "DFF" | "DFFSR" | "$__RAMGEM_SYNC_");
1349            let is_sky130_seq =
1350                is_sky130_cell(celltype) && is_sequential_cell(extract_cell_type(celltype));
1351
1352            if !is_aigpdk_seq && !is_sky130_seq {
1353                continue;
1354            }
1355
1356            seq_count += 1;
1357            if seq_count <= 5 {
1358                use netlistdb::GeneralHierName;
1359                clilog::info!(
1360                    "  Processing seq cell {} ({}): {}",
1361                    seq_count,
1362                    celltype,
1363                    netlistdb.cellnames[cellid].dbg_fmt_hier()
1364                );
1365            } else if seq_count % 500 == 0 {
1366                clilog::info!(
1367                    "  Processed {} sequential cells ({} clock traces) in {:?}...",
1368                    seq_count,
1369                    trace_count,
1370                    clock_start.elapsed()
1371                );
1372            }
1373            for pinid in netlistdb.cell2pin.iter_set(cellid) {
1374                let pin_name = netlistdb.pinnames[pinid].1.as_str();
1375                // Both AIGPDK and SKY130 use "CLK" for clock pins
1376                if !matches!(pin_name, "CLK" | "PORT_R_CLK" | "PORT_W_CLK") {
1377                    continue;
1378                }
1379                trace_count += 1;
1380                if seq_count <= 5 {
1381                    clilog::info!("    Tracing clock pin {} for cell {}...", pinid, cellid);
1382                }
1383                if let Err(pinid) = aig.trace_clock_pin(netlistdb, pinid, false, true) {
1384                    use netlistdb::GeneralHierName;
1385                    panic!(
1386                        "Tracing clock pin of cell {} error: \
1387                            there is a multi-input cell driving {} \
1388                            that clocks this sequential element. \
1389                            Clock gating need to be manually patched atm.",
1390                        netlistdb.cellnames[cellid].dbg_fmt_hier(),
1391                        netlistdb.pinnames[pinid].dbg_fmt_pin()
1392                    );
1393                }
1394            }
1395        }
1396        clilog::info!(
1397            "Clock tracing done in {:?}, {} sequential cells",
1398            clock_start.elapsed(),
1399            seq_count
1400        );
1401
1402        for (&clk, &(flagr, flagf)) in &aig.clock_pin2aigpins {
1403            clilog::info!(
1404                "inferred clock port {} ({})",
1405                netlistdb.pinnames[clk].dbg_fmt_pin(),
1406                match (flagr, flagf) {
1407                    (_, usize::MAX) => "posedge",
1408                    (usize::MAX, _) => "negedge",
1409                    _ => "posedge & negedge",
1410                }
1411            );
1412        }
1413
1414        clilog::info!("Starting AIG DFS build for {} pins...", netlistdb.num_pins);
1415        let dfs_start = std::time::Instant::now();
1416
1417        let mut topo_vis = vec![false; netlistdb.num_pins];
1418        let mut topo_instack = vec![false; netlistdb.num_pins];
1419
1420        for pinid in 0..netlistdb.num_pins {
1421            aig.dfs_netlistdb_build_aig(
1422                netlistdb,
1423                &mut topo_vis,
1424                &mut topo_instack,
1425                pinid,
1426                pdk_models,
1427            );
1428        }
1429
1430        clilog::info!(
1431            "AIG DFS build done in {:?}, {} AIG pins created",
1432            dfs_start.elapsed(),
1433            aig.num_aigpins
1434        );
1435
1436        for cellid in 0..netlistdb.num_cells {
1437            let celltype = netlistdb.celltypes[cellid].as_str();
1438
1439            if matches!(celltype, "DFF" | "DFFSR") {
1440                let mut ap_s_iv = 1;
1441                let mut ap_r_iv = 1;
1442                let mut ap_d_iv = 0;
1443                let mut ap_clken_iv = 0;
1444                for pinid in netlistdb.cell2pin.iter_set(cellid) {
1445                    let pin_iv = aig.pin2aigpin_iv[pinid];
1446                    match netlistdb.pinnames[pinid].1.as_str() {
1447                        "D" => ap_d_iv = pin_iv,
1448                        "S" => ap_s_iv = pin_iv,
1449                        "R" => ap_r_iv = pin_iv,
1450                        "CLK" => {
1451                            ap_clken_iv =
1452                                aig.trace_clock_pin(netlistdb, pinid, false, false).unwrap()
1453                        }
1454                        _ => {}
1455                    }
1456                }
1457                let mut d_in = ap_d_iv;
1458
1459                // AIGPDK DFFSR .lib says: clear="(!R)", preset="(!S)"
1460                // i.e. R=0 â†’ Q=0 (clear), S=0 â†’ Q=1 (preset)
1461                // Same active-low semantics as SKY130 RESET_B/SET_B:
1462                //   d_in = AND(OR(D, !S), R)
1463                //   en_iv = OR(posedge, !R, !S)  (latch whenever R or S is active)
1464                d_in = aig.add_and_gate(d_in ^ 1, ap_s_iv) ^ 1;
1465                ap_clken_iv = aig.add_and_gate(ap_clken_iv ^ 1, ap_s_iv) ^ 1;
1466                d_in = aig.add_and_gate(d_in, ap_r_iv);
1467                ap_clken_iv = aig.add_and_gate(ap_clken_iv ^ 1, ap_r_iv) ^ 1;
1468                let dff = aig.dffs.entry(cellid).or_default();
1469                dff.en_iv = ap_clken_iv;
1470                dff.d_iv = d_in;
1471                assert_ne!(dff.q, 0);
1472            } else if is_sky130_cell(celltype) && is_sequential_cell(extract_cell_type(celltype)) {
1473                // Handle SKY130 DFFs (dfxtp, edfxtp, dfrtp, etc.)
1474                let cell_type = extract_cell_type(celltype);
1475                let mut ap_d_iv = 0;
1476                let mut ap_clken_iv = 0;
1477                let mut ap_enable_iv = 1; // For edfxtp (data enable)
1478                let mut ap_s_iv = 1; // Default: no set (SET_B=1, inactive)
1479                let mut ap_r_iv = 1; // Default: no reset (RESET_B=1, inactive)
1480
1481                for pinid in netlistdb.cell2pin.iter_set(cellid) {
1482                    let pin_iv = aig.pin2aigpin_iv[pinid];
1483                    match netlistdb.pinnames[pinid].1.as_str() {
1484                        "D" => ap_d_iv = pin_iv,
1485                        "DE" => ap_enable_iv = pin_iv, // Data enable for edfxtp
1486                        "SET_B" => ap_s_iv = pin_iv, // Active-low: AND/OR formulas handle it directly
1487                        "RESET_B" => ap_r_iv = pin_iv, // Active-low: AND(d, RESET_B) â†’ RESET_B=0 forces d=0
1488                        "CLK" => {
1489                            ap_clken_iv =
1490                                aig.trace_clock_pin(netlistdb, pinid, false, false).unwrap()
1491                        }
1492                        _ => {}
1493                    }
1494                }
1495
1496                // For edfxtp, the clock enable is gated by DE (data enable)
1497                if cell_type == "edfxtp" {
1498                    ap_clken_iv = aig.add_and_gate(ap_clken_iv, ap_enable_iv);
1499                }
1500
1501                let mut d_in = ap_d_iv;
1502
1503                // Apply async set/reset to D input and clock enable (same as AIGPDK DFF)
1504                d_in = aig.add_and_gate(d_in ^ 1, ap_s_iv) ^ 1;
1505                ap_clken_iv = aig.add_and_gate(ap_clken_iv ^ 1, ap_s_iv) ^ 1;
1506                d_in = aig.add_and_gate(d_in, ap_r_iv);
1507                ap_clken_iv = aig.add_and_gate(ap_clken_iv ^ 1, ap_r_iv) ^ 1;
1508
1509                let dff = aig.dffs.entry(cellid).or_default();
1510                dff.en_iv = ap_clken_iv;
1511                dff.d_iv = d_in;
1512                assert_ne!(dff.q, 0, "SKY130 DFF {} has no Q output built", cellid);
1513            } else if celltype == "$__RAMGEM_SYNC_" {
1514                let mut sram = aig.srams.entry(cellid).or_default().clone();
1515                let mut write_clken_iv = 0;
1516                for pinid in netlistdb.cell2pin.iter_set(cellid) {
1517                    let bit = netlistdb.pinnames[pinid].2.map(|i| i as usize);
1518                    let pin_iv = aig.pin2aigpin_iv[pinid];
1519                    match netlistdb.pinnames[pinid].1.as_str() {
1520                        "PORT_R_ADDR" => {
1521                            sram.port_r_addr_iv[bit.unwrap()] = pin_iv;
1522                        }
1523                        "PORT_R_CLK" => {
1524                            sram.port_r_en_iv =
1525                                aig.trace_clock_pin(netlistdb, pinid, false, false).unwrap();
1526                        }
1527                        "PORT_W_ADDR" => {
1528                            sram.port_w_addr_iv[bit.unwrap()] = pin_iv;
1529                        }
1530                        "PORT_W_CLK" => {
1531                            write_clken_iv =
1532                                aig.trace_clock_pin(netlistdb, pinid, false, false).unwrap();
1533                        }
1534                        "PORT_W_WR_DATA" => {
1535                            sram.port_w_wr_data_iv[bit.unwrap()] = pin_iv;
1536                        }
1537                        "PORT_W_WR_EN" => {
1538                            sram.port_w_wr_en_iv[bit.unwrap()] = pin_iv;
1539                        }
1540                        _ => {}
1541                    }
1542                }
1543                for i in 0..32 {
1544                    let or_en = sram.port_w_wr_en_iv[i];
1545                    let or_en = aig.add_and_gate(or_en, write_clken_iv);
1546                    sram.port_w_wr_en_iv[i] = or_en;
1547                }
1548                *aig.srams.get_mut(&cellid).unwrap() = sram;
1549            } else if celltype.starts_with("CF_SRAM_") {
1550                // ChipFlow SRAM: single-port with shared address
1551                // Pins: CLKin, EN, R_WB, AD[9:0], BEN[31:0], DI[31:0], DO[31:0]
1552                let mut sram = aig.srams.entry(cellid).or_default().clone();
1553                let mut clken_iv = 0;
1554                let mut r_wb_iv = 0; // 0=write, 1=read
1555
1556                for pinid in netlistdb.cell2pin.iter_set(cellid) {
1557                    let bit = netlistdb.pinnames[pinid].2.map(|i| i as usize);
1558                    let pin_iv = aig.pin2aigpin_iv[pinid];
1559                    match netlistdb.pinnames[pinid].1.as_str() {
1560                        "CLKin" => {
1561                            clken_iv = aig.trace_clock_pin(netlistdb, pinid, false, false).unwrap();
1562                        }
1563                        "EN" => {
1564                            // EN combined with clock for read enable
1565                            // (we'll combine later)
1566                        }
1567                        "R_WB" => {
1568                            r_wb_iv = pin_iv; // 0=write, 1=read
1569                        }
1570                        "AD" => {
1571                            // Shared address for read and write
1572                            let idx = bit.unwrap();
1573                            if idx < AIGPDK_SRAM_ADDR_WIDTH {
1574                                sram.port_r_addr_iv[idx] = pin_iv;
1575                                sram.port_w_addr_iv[idx] = pin_iv;
1576                            }
1577                        }
1578                        "DI" => {
1579                            sram.port_w_wr_data_iv[bit.unwrap()] = pin_iv;
1580                        }
1581                        "BEN" => {
1582                            // Byte enable becomes write enable when R_WB=0
1583                            sram.port_w_wr_en_iv[bit.unwrap()] = pin_iv;
1584                        }
1585                        _ => {} // Ignore control pins like SM, TM, Scan*, vpwr*
1586                    }
1587                }
1588
1589                // Read enable: clken AND R_WB (read when R_WB=1)
1590                sram.port_r_en_iv = aig.add_and_gate(clken_iv, r_wb_iv);
1591
1592                // Write enable: clken AND !R_WB (write when R_WB=0) AND BEN
1593                let write_clken_iv = aig.add_and_gate(clken_iv, r_wb_iv ^ 1);
1594                for i in 0..32 {
1595                    let ben_iv = sram.port_w_wr_en_iv[i];
1596                    sram.port_w_wr_en_iv[i] = aig.add_and_gate(write_clken_iv, ben_iv);
1597                }
1598
1599                *aig.srams.get_mut(&cellid).unwrap() = sram;
1600            } else if netlistdb.celltypes[cellid].as_str() == "GEM_ASSERT" {
1601                // Parse GEM_ASSERT cells for assertion checking
1602                // GEM_ASSERT has: CLK (trigger), EN (enable), A (condition)
1603                // Assertion fails when EN is high and A is low
1604                let mut ap_en_iv = 1; // Default: always enabled
1605                let mut ap_a_iv = 1; // Default: always passing
1606                let mut ap_clken_iv = 1; // Default: always triggered
1607
1608                for pinid in netlistdb.cell2pin.iter_set(cellid) {
1609                    let pin_iv = aig.pin2aigpin_iv[pinid];
1610                    match netlistdb.pinnames[pinid].1.as_str() {
1611                        "EN" => ap_en_iv = pin_iv,
1612                        "A" => ap_a_iv = pin_iv,
1613                        "CLK" => {
1614                            // Try to trace clock, but if it's tied to 1, use constant
1615                            if let Ok(clken) = aig.trace_clock_pin(netlistdb, pinid, false, false) {
1616                                ap_clken_iv = clken;
1617                            }
1618                        }
1619                        _ => {}
1620                    }
1621                }
1622
1623                // The condition for firing an assertion event:
1624                // fire = clk_enable && EN && !A
1625                // We store the "fire" condition (inverted A ANDed with EN and clock)
1626                let fire_condition = aig.add_and_gate(ap_en_iv, ap_a_iv ^ 1);
1627                let fire_condition = aig.add_and_gate(fire_condition, ap_clken_iv);
1628
1629                let simcontrol = aig.simcontrols.entry(cellid).or_default();
1630                simcontrol.condition_iv = fire_condition;
1631                simcontrol.control_type = None; // Assertion, not $stop/$finish
1632                simcontrol.message_id = cellid as u32; // Use cell ID as message ID for now
1633
1634                clilog::debug!(
1635                    "Found GEM_ASSERT cell {} (condition_iv={}, en_iv={}, a_iv={}, clken_iv={})",
1636                    cellid,
1637                    fire_condition,
1638                    ap_en_iv,
1639                    ap_a_iv,
1640                    ap_clken_iv
1641                );
1642            } else if netlistdb.celltypes[cellid].as_str() == "GEM_DISPLAY" {
1643                // Parse GEM_DISPLAY cells for $display/$write support
1644                // GEM_DISPLAY has: CLK (trigger), EN (enable), MSG_ID[31:0] (argument values)
1645                // Plus attributes: gem_format (format string), gem_args_width (arg width)
1646                let mut dp_en_iv = 1; // Default: always enabled
1647                let mut dp_clken_iv = 1; // Default: always triggered
1648                let mut dp_args_iv = Vec::new();
1649
1650                for pinid in netlistdb.cell2pin.iter_set(cellid) {
1651                    let pin_iv = aig.pin2aigpin_iv[pinid];
1652                    match netlistdb.pinnames[pinid].1.as_str() {
1653                        "EN" => dp_en_iv = pin_iv,
1654                        "CLK" => {
1655                            // Try to trace clock for edge detection
1656                            if let Ok(clken) = aig.trace_clock_pin(netlistdb, pinid, false, false) {
1657                                dp_clken_iv = clken;
1658                            }
1659                        }
1660                        "MSG_ID" => {
1661                            // MSG_ID is a 32-bit bus, collect all bit pins
1662                            dp_args_iv.push(pin_iv);
1663                        }
1664                        _ => {}
1665                    }
1666                }
1667
1668                // The condition for firing a display event:
1669                // fire = clk_enable && EN
1670                let fire_condition = aig.add_and_gate(dp_en_iv, dp_clken_iv);
1671
1672                // Get cell name for matching with JSON attributes later
1673                use netlistdb::GeneralHierName;
1674                let cell_name = netlistdb.cellnames[cellid].dbg_fmt_hier();
1675
1676                // Create DisplayNode
1677                let display = aig.displays.entry(cellid).or_default();
1678                display.enable_iv = fire_condition;
1679                display.clken_iv = dp_clken_iv;
1680                // Format string will be populated from JSON attributes later
1681                display.format = format!("$display at cell {}", cellid);
1682                display.args_iv = dp_args_iv;
1683                display.arg_widths = vec![32]; // Default: 32-bit argument
1684                display.cell_name = cell_name.to_string();
1685
1686                clilog::debug!(
1687                    "Found GEM_DISPLAY cell {} '{}' (enable_iv={}, clken_iv={}, args={})",
1688                    cellid,
1689                    display.cell_name,
1690                    fire_condition,
1691                    dp_clken_iv,
1692                    display.args_iv.len()
1693                );
1694            }
1695        }
1696
1697        // Validate AIG: check for bad operand values
1698        for (i, driver) in aig.drivers.iter().enumerate() {
1699            if let DriverType::AndGate(a, b) = *driver {
1700                if (a >> 1) > aig.num_aigpins {
1701                    panic!("AIG validation: AND gate at pin {} has bad operand a={} (a>>1={}, num_aigpins={})", i, a, a >> 1, aig.num_aigpins);
1702                }
1703                if (b >> 1) > aig.num_aigpins {
1704                    panic!("AIG validation: AND gate at pin {} has bad operand b={} (b>>1={}, num_aigpins={})", i, b, b >> 1, aig.num_aigpins);
1705                }
1706            }
1707        }
1708        // Validate DFF d_iv and en_iv
1709        for (cellid, dff) in &aig.dffs {
1710            if (dff.d_iv >> 1) > aig.num_aigpins {
1711                use netlistdb::GeneralHierName;
1712                let cell_name = netlistdb.cellnames[*cellid].dbg_fmt_hier();
1713                panic!(
1714                    "AIG validation: DFF {} ({}) has bad d_iv={} (d_iv>>1={})",
1715                    cellid,
1716                    cell_name,
1717                    dff.d_iv,
1718                    dff.d_iv >> 1
1719                );
1720            }
1721            if (dff.en_iv >> 1) > aig.num_aigpins {
1722                use netlistdb::GeneralHierName;
1723                let cell_name = netlistdb.cellnames[*cellid].dbg_fmt_hier();
1724                panic!(
1725                    "AIG validation: DFF {} ({}) has bad en_iv={} (en_iv>>1={})",
1726                    cellid,
1727                    cell_name,
1728                    dff.en_iv,
1729                    dff.en_iv >> 1
1730                );
1731            }
1732        }
1733        // Validate primary_outputs
1734        for &po_iv in &aig.primary_outputs {
1735            if (po_iv >> 1) > aig.num_aigpins {
1736                // Find which pin has this value
1737                for pinid in 0..netlistdb.num_pins {
1738                    if aig.pin2aigpin_iv[pinid] == po_iv {
1739                        use netlistdb::GeneralHierName;
1740                        let pin_name = netlistdb.pinnames[pinid].dbg_fmt_pin();
1741                        let cellid = netlistdb.pin2cell[pinid];
1742                        let cell_name = netlistdb.cellnames[cellid].dbg_fmt_hier();
1743                        let cell_type = &netlistdb.celltypes[cellid];
1744                        let dir = netlistdb.pindirect[pinid];
1745                        panic!("AIG validation: primary output has bad value iv={} from pin {} (cell {} type={}, dir={:?})", po_iv, pin_name, cell_name, cell_type, dir);
1746                    }
1747                }
1748                panic!("AIG validation: primary output has bad value iv={} (iv>>1={}) - could not find source pin", po_iv, po_iv >> 1);
1749            }
1750        }
1751
1752        aig.fanouts_start = vec![0; aig.num_aigpins + 2];
1753        for (_i, driver) in aig.drivers.iter().enumerate() {
1754            if let DriverType::AndGate(a, b) = *driver {
1755                if (a >> 1) != 0 {
1756                    aig.fanouts_start[a >> 1] += 1;
1757                }
1758                if (b >> 1) != 0 {
1759                    aig.fanouts_start[b >> 1] += 1;
1760                }
1761            }
1762        }
1763        for i in 1..aig.num_aigpins + 2 {
1764            aig.fanouts_start[i] += aig.fanouts_start[i - 1];
1765        }
1766        aig.fanouts = vec![0; aig.fanouts_start[aig.num_aigpins + 1]];
1767        for (i, driver) in aig.drivers.iter().enumerate() {
1768            if let DriverType::AndGate(a, b) = *driver {
1769                if (a >> 1) != 0 {
1770                    let st = aig.fanouts_start[a >> 1] - 1;
1771                    aig.fanouts_start[a >> 1] = st;
1772                    aig.fanouts[st] = i;
1773                }
1774                if (b >> 1) != 0 {
1775                    let st = aig.fanouts_start[b >> 1] - 1;
1776                    aig.fanouts_start[b >> 1] = st;
1777                    aig.fanouts[st] = i;
1778                }
1779            }
1780        }
1781
1782        aig
1783    }

1785    pub fn topo_traverse_generic(
1786        &self,
1787        endpoints: Option<&Vec<usize>>,
1788        is_primary_input: Option<&IndexSet<usize>>,
1789    ) -> Vec<usize> {
1790        let mut vis = IndexSet::new();
1791        let mut ret = Vec::new();
1792        fn dfs_topo(
1793            aig: &AIG,
1794            vis: &mut IndexSet<usize>,
1795            ret: &mut Vec<usize>,
1796            is_primary_input: Option<&IndexSet<usize>>,
1797            u: usize,
1798        ) {
1799            if vis.contains(&u) {
1800                return;
1801            }
1802            vis.insert(u);
1803            if let DriverType::AndGate(a, b) = aig.drivers[u] {
1804                if is_primary_input.map(|s| s.contains(&u)) != Some(true) {
1805                    if (a >> 1) != 0 {
1806                        dfs_topo(aig, vis, ret, is_primary_input, a >> 1);
1807                    }
1808                    if (b >> 1) != 0 {
1809                        dfs_topo(aig, vis, ret, is_primary_input, b >> 1);
1810                    }
1811                }
1812            }
1813            ret.push(u);
1814        }
1815        if let Some(endpoints) = endpoints {
1816            for &endpoint in endpoints {
1817                dfs_topo(self, &mut vis, &mut ret, is_primary_input, endpoint);
1818            }
1819        } else {
1820            for i in 1..self.num_aigpins + 1 {
1821                dfs_topo(self, &mut vis, &mut ret, is_primary_input, i);
1822            }
1823        }
1824        ret
1825    }

1878    /// Load timing information from a Liberty library.
1879    ///
1880    /// This initializes the gate_delays vector and dff_timing field.
1881    pub fn load_timing_library(&mut self, lib: &crate::liberty_parser::TimingLibrary) {
1882        // Initialize gate delays for all AIG pins
1883        // Index 0 is Tie0 which has zero delay
1884        self.gate_delays = vec![(0, 0); self.num_aigpins + 1];
1885
1886        // Get default AND gate delay (all variants have same timing in AIGPDK)
1887        let and_delay = lib.and_gate_delay("AND2_00_0").unwrap_or((1, 1));
1888
1889        // Assign delays based on driver type
1890        for i in 1..=self.num_aigpins {
1891            let delay = match &self.drivers[i] {
1892                DriverType::AndGate(_, _) => and_delay,
1893                DriverType::InputPort(_) => (0, 0), // Primary inputs have zero arrival
1894                DriverType::InputClockFlag(_, _) => (0, 0), // Clock flags
1895                DriverType::DFF(_) => {
1896                    // Clock-to-Q delay
1897                    lib.dff_timing()
1898                        .map(|t| (t.clk_to_q_rise_ps, t.clk_to_q_fall_ps))
1899                        .unwrap_or((0, 0))
1900                }
1901                DriverType::SRAM(_) => {
1902                    // SRAM read delay
1903                    lib.sram_timing()
1904                        .map(|t| (t.read_clk_to_data_rise_ps, t.read_clk_to_data_fall_ps))
1905                        .unwrap_or((1, 1))
1906                }
1907                DriverType::Tie0 => (0, 0),
1908            };
1909            self.gate_delays[i] = delay;
1910        }
1911
1912        // Load DFF timing constraints
1913        self.dff_timing = lib.dff_timing();
1914
1915        clilog::info!(
1916            "Loaded timing library: AND delay={}ps, DFF setup={}ps, hold={}ps",
1917            and_delay.0,
1918            self.dff_timing.as_ref().map(|t| t.max_setup()).unwrap_or(0),
1919            self.dff_timing.as_ref().map(|t| t.max_hold()).unwrap_or(0)
1920        );
1921    }

1923    /// Compute static timing analysis (STA) for the AIG.
1924    ///
1925    /// This computes arrival times at all nodes and calculates setup/hold
1926    /// slacks for all DFFs. Must call `load_timing_library` first.
1927    ///
1928    /// Returns a `TimingReport` with summary statistics.
1929    pub fn compute_timing(&mut self) -> TimingReport {
1930        // Initialize arrival times (min, max) for all pins
1931        self.arrival_times = vec![(0, 0); self.num_aigpins + 1];
1932
1933        // Get topological order (already guaranteed in AIG)
1934        // Traverse in order 1..num_aigpins since drivers are in topo order
1935        for i in 1..=self.num_aigpins {
1936            let (min_arrival, max_arrival) = match &self.drivers[i] {
1937                DriverType::AndGate(a, b) => {
1938                    let (delay_rise, delay_fall) = self.gate_delays[i];
1939                    let delay = delay_rise.max(delay_fall);
1940
1941                    // Get arrival times of inputs (strip inversion bit)
1942                    let a_idx = a >> 1;
1943                    let b_idx = b >> 1;
1944
1945                    let (a_min, a_max) = if a_idx == 0 {
1946                        (0, 0)
1947                    } else {
1948                        self.arrival_times[a_idx]
1949                    };
1950                    let (b_min, b_max) = if b_idx == 0 {
1951                        (0, 0)
1952                    } else {
1953                        self.arrival_times[b_idx]
1954                    };
1955
1956                    // Min arrival: min of inputs + delay
1957                    // Max arrival: max of inputs + delay
1958                    (a_min.min(b_min) + delay, a_max.max(b_max) + delay)
1959                }
1960                DriverType::InputPort(_) | DriverType::InputClockFlag(_, _) | DriverType::Tie0 => {
1961                    // Primary inputs arrive at time 0
1962                    (0, 0)
1963                }
1964                DriverType::DFF(_) => {
1965                    // DFF output arrives after clock-to-Q delay
1966                    let (delay_rise, delay_fall) = self.gate_delays[i];
1967                    (delay_rise.max(delay_fall), delay_rise.max(delay_fall))
1968                }
1969                DriverType::SRAM(_) => {
1970                    // SRAM read data arrives after read delay
1971                    let (delay_rise, delay_fall) = self.gate_delays[i];
1972                    (delay_rise.max(delay_fall), delay_rise.max(delay_fall))
1973                }
1974            };
1975
1976            self.arrival_times[i] = (min_arrival, max_arrival);
1977        }
1978
1979        // Compute setup and hold slacks for each DFF
1980        self.setup_slacks = Vec::with_capacity(self.dffs.len());
1981        self.hold_slacks = Vec::with_capacity(self.dffs.len());
1982
1983        let dff_timing = self.dff_timing.clone().unwrap_or_default();
1984        let setup_time = dff_timing.max_setup();
1985        let hold_time = dff_timing.max_hold();
1986
1987        let mut report = TimingReport::default();
1988        report.num_endpoints = self.dffs.len();
1989
1990        for (_cell_id, dff) in &self.dffs {
1991            // Get data arrival time at D input
1992            let d_idx = dff.d_iv >> 1;
1993            let (_, data_max_arrival) = if d_idx == 0 {
1994                (0, 0)
1995            } else {
1996                self.arrival_times[d_idx]
1997            };
1998
1999            // Clock arrival is at time 0 (or clock_period for setup check)
2000            // Setup check: data must arrive before clock_period - setup_time
2001            let setup_slack =
2002                (self.clock_period_ps as i64) - (data_max_arrival as i64) - (setup_time as i64);
2003
2004            // Hold check: data must be stable for hold_time after clock
2005            // For hold, we check min arrival vs hold_time
2006            let (data_min_arrival, _) = if d_idx == 0 {
2007                (0, 0)
2008            } else {
2009                self.arrival_times[d_idx]
2010            };
2011            let hold_slack = (data_min_arrival as i64) - (hold_time as i64);
2012
2013            self.setup_slacks.push(setup_slack);
2014            self.hold_slacks.push(hold_slack);
2015
2016            // Update report statistics
2017            report.worst_setup_slack = report.worst_setup_slack.min(setup_slack);
2018            report.worst_hold_slack = report.worst_hold_slack.min(hold_slack);
2019
2020            if setup_slack < 0 {
2021                report.setup_violations += 1;
2022            }
2023            if hold_slack < 0 {
2024                report.hold_violations += 1;
2025            }
2026        }
2027
2028        // Find critical path (longest arrival time at any endpoint)
2029        for &po_iv in &self.primary_outputs {
2030            let po = po_iv >> 1; // Strip inversion bit
2031            if po > 0 && po <= self.num_aigpins {
2032                let (_, max_arr) = self.arrival_times[po];
2033                report.critical_path_delay = report.critical_path_delay.max(max_arr);
2034            }
2035        }
2036        for (_cell_id, dff) in &self.dffs {
2037            let d_idx = dff.d_iv >> 1;
2038            if d_idx > 0 {
2039                let (_, max_arr) = self.arrival_times[d_idx];
2040                report.critical_path_delay = report.critical_path_delay.max(max_arr);
2041            }
2042        }
2043
2044        report
2045    }

2113/// A reusable topological traverser with dense visited buffer.
2114///
2115/// Uses a generation counter pattern to avoid clearing the visited buffer
2116/// between traversals. This is significantly faster than `IndexSet<usize>`
2117/// for repeated traversals on the same AIG.
2118pub struct TopoTraverser {
2119    visited: Vec<u32>,
2120    generation: u32,
2121}
2122
2123impl TopoTraverser {
2124    /// Create a new traverser for an AIG with the given number of pins.
2125    pub fn new(num_aigpins: usize) -> Self {
2126        Self {
2127            visited: vec![0; num_aigpins + 1],
2128            generation: 1,
2129        }
2130    }
2131
2132    /// Reset the generation counter, preparing for a new traversal.
2133    fn new_generation(&mut self) {
2134        self.generation = self.generation.wrapping_add(1);
2135        if self.generation == 0 {
2136            // Overflow: clear the buffer and reset
2137            self.visited.fill(0);
2138            self.generation = 1;
2139        }
2140    }
2141
2142    #[inline(always)]
2143    fn is_visited(&self, u: usize) -> bool {
2144        self.visited[u] == self.generation
2145    }
2146
2147    #[inline(always)]
2148    fn mark_visited(&mut self, u: usize) {
2149        self.visited[u] = self.generation;
2150    }
2151
2152    /// Perform a topological traversal, equivalent to `AIG::topo_traverse_generic`.
2153    ///
2154    /// Uses an iterative stack-based DFS with a dense visited buffer for speed.
2155    pub fn topo_traverse(
2156        &mut self,
2157        aig: &AIG,
2158        endpoints: Option<&Vec<usize>>,
2159        is_primary_input: Option<&IndexSet<usize>>,
2160    ) -> Vec<usize> {
2161        self.new_generation();
2162        let mut ret = Vec::new();
2163        // Two-phase iterative DFS: Visit checks + pushes deps, Process emits
2164        enum Phase {
2165            Visit(usize),
2166            Process(usize),
2167        }
2168        let mut stack = Vec::new();
2169
2170        if let Some(endpoints) = endpoints {
2171            for &endpoint in endpoints {
2172                stack.push(Phase::Visit(endpoint));
2173            }
2174        } else {
2175            for i in (1..aig.num_aigpins + 1).rev() {
2176                stack.push(Phase::Visit(i));
2177            }
2178        }
2179
2180        while let Some(item) = stack.pop() {
2181            match item {
2182                Phase::Visit(u) => {
2183                    if self.is_visited(u) {
2184                        continue;
2185                    }
2186                    self.mark_visited(u);
2187                    stack.push(Phase::Process(u));
2188                    if let DriverType::AndGate(a, b) = aig.drivers[u] {
2189                        if is_primary_input.map(|s| s.contains(&u)) != Some(true) {
2190                            // Push b first so a is processed first (stack order)
2191                            if (b >> 1) != 0 {
2192                                stack.push(Phase::Visit(b >> 1));
2193                            }
2194                            if (a >> 1) != 0 {
2195                                stack.push(Phase::Visit(a >> 1));
2196                            }
2197                        }
2198                    }
2199                }
2200                Phase::Process(u) => {
2201                    ret.push(u);
2202                }
2203            }
2204        }
2205        ret
2206    }
2207
2208    /// Perform a topological traversal and also produce a bitset of visited nodes.
2209    ///
2210    /// Returns `(order, bitset)` where `bitset` has bit `node` set for each visited node.
2211    /// The bitset is `Vec<u64>` with `(num_aigpins + 64) / 64` words.
2212    pub fn topo_traverse_with_bitset(
2213        &mut self,
2214        aig: &AIG,
2215        endpoints: Option<&Vec<usize>>,
2216        is_primary_input: Option<&IndexSet<usize>>,
2217    ) -> (Vec<usize>, Vec<u64>) {
2218        let order = self.topo_traverse(aig, endpoints, is_primary_input);
2219        let num_words = (aig.num_aigpins + 64) / 64;
2220        let mut bitset = vec![0u64; num_words];
2221        for &node in &order {
