3//! SKY130 standard cell library support for GEM.
4//!
5//! This module provides pin direction and width information for the SKY130
6//! open-source PDK standard cells. SKY130 cells follow the naming convention:
7//! `sky130_fd_sc_<variant>__<celltype>_<drive>`
8//!
9//! Supported variants: hd (high-density), hs (high-speed), ms (medium-speed)
10
11use compact_str::CompactString;
12use netlistdb::{Direction, LeafPinProvider};
13use sverilogparse::SVerilogRange;
14
15/// LeafPinProvider implementation for SKY130 standard cells.
16pub struct SKY130LeafPins;
17
18impl LeafPinProvider for SKY130LeafPins {
19    fn direction_of(
20        &self,
21        macro_name: &CompactString,
22        pin_name: &CompactString,
23        _pin_idx: Option<isize>,
24    ) -> Direction {
25        let cell_type = extract_cell_type(macro_name);
26        let pin = pin_name.as_str();
27
28        match cell_type {
29            // Inverters: A -> Y
30            t if t.starts_with("inv") => match pin {
31                "A" => Direction::I,
32                "Y" => Direction::O,
33                _ => unknown_pin(macro_name, pin_name),
34            },
35
36            // Buffers: A -> X
37            t if t.starts_with("buf") || t.starts_with("clkbuf") || t.starts_with("clkdlybuf") => {
38                match pin {
39                    "A" => Direction::I,
40                    "X" => Direction::O,
41                    _ => unknown_pin(macro_name, pin_name),
42                }
43            }
44
45            // Buffer-inverter: A -> Y
46            t if t.starts_with("bufinv") => match pin {
47                "A" => Direction::I,
48                "Y" => Direction::O,
49                _ => unknown_pin(macro_name, pin_name),
50            },
51
52            // Clock inverters: A -> Y
53            t if t.starts_with("clkinv") => match pin {
54                "A" => Direction::I,
55                "Y" => Direction::O,
56                _ => unknown_pin(macro_name, pin_name),
57            },
58
59            // Delay gate: A -> X
60            t if t.starts_with("dlygate") => match pin {
61                "A" => Direction::I,
62                "X" => Direction::O,
63                _ => unknown_pin(macro_name, pin_name),
64            },
65
66            // Low-power isolation buffers: A, SLEEP/SLEEP_B -> X
67            t if t.starts_with("lpflow_isobufsrc") => match pin {
68                "A" | "SLEEP" => Direction::I,
69                "X" => Direction::O,
70                _ => unknown_pin(macro_name, pin_name),
71            },
72            t if t.starts_with("lpflow_inputiso") => match pin {
73                "A" | "SLEEP" | "SLEEP_B" => Direction::I,
74                "X" => Direction::O,
75                _ => unknown_pin(macro_name, pin_name),
76            },
77
78            // 2-input AND: A, B -> X (and2b has A_N or B_N)
79            t if t.starts_with("and2") => match pin {
80                "A" | "A_N" | "B" | "B_N" => Direction::I,
81                "X" => Direction::O,
82                _ => unknown_pin(macro_name, pin_name),
83            },
84
85            // 3-input AND: A, B, C -> X (and3b has _N variants)
86            t if t.starts_with("and3") => match pin {
87                "A" | "A_N" | "B" | "B_N" | "C" | "C_N" => Direction::I,
88                "X" => Direction::O,
89                _ => unknown_pin(macro_name, pin_name),
90            },
91
92            // 4-input AND: A, B, C, D -> X (and4b/and4bb have _N variants)
93            t if t.starts_with("and4") => match pin {
94                "A" | "A_N" | "B" | "B_N" | "C" | "C_N" | "D" | "D_N" => Direction::I,
95                "X" => Direction::O,
96                _ => unknown_pin(macro_name, pin_name),
97            },
98
99            // 2-input OR: A, B -> X (or2b has _N variants)
100            t if t.starts_with("or2") => match pin {
101                "A" | "A_N" | "B" | "B_N" => Direction::I,
102                "X" => Direction::O,
103                _ => unknown_pin(macro_name, pin_name),
104            },
105
106            // 3-input OR: A, B, C -> X (or3b has C_N instead of C)
107            t if t.starts_with("or3") => match pin {
108                "A" | "A_N" | "B" | "B_N" | "C" | "C_N" => Direction::I,
109                "X" => Direction::O,
110                _ => unknown_pin(macro_name, pin_name),
111            },
112
113            // 4-input OR: A, B, C, D -> X (or4b/or4bb have _N variants)
114            t if t.starts_with("or4") => match pin {
115                "A" | "A_N" | "B" | "B_N" | "C" | "C_N" | "D" | "D_N" => Direction::I,
116                "X" => Direction::O,
117                _ => unknown_pin(macro_name, pin_name),
118            },
119
120            // 2-input NAND: A, B -> Y (nand2b has A_N or B_N)
121            t if t.starts_with("nand2") => match pin {
122                "A" | "A_N" | "B" | "B_N" => Direction::I,
123                "Y" => Direction::O,
124                _ => unknown_pin(macro_name, pin_name),
125            },
126
127            // 3-input NAND: A, B, C -> Y (nand3b has A_N)
128            t if t.starts_with("nand3") => match pin {
129                "A" | "A_N" | "B" | "B_N" | "C" | "C_N" => Direction::I,
130                "Y" => Direction::O,
131                _ => unknown_pin(macro_name, pin_name),
132            },
133
134            // 4-input NAND: A, B, C, D -> Y (nand4b/nand4bb have _N variants)
135            t if t.starts_with("nand4") => match pin {
136                "A" | "A_N" | "B" | "B_N" | "C" | "C_N" | "D" | "D_N" => Direction::I,
137                "Y" => Direction::O,
138                _ => unknown_pin(macro_name, pin_name),
139            },
140
141            // 2-input NOR: A, B -> Y (nor2b has A_N or B_N)
142            t if t.starts_with("nor2") => match pin {
143                "A" | "A_N" | "B" | "B_N" => Direction::I,
144                "Y" => Direction::O,
145                _ => unknown_pin(macro_name, pin_name),
146            },
147
148            // 3-input NOR: A, B, C -> Y (nor3b has C_N)
149            t if t.starts_with("nor3") => match pin {
150                "A" | "A_N" | "B" | "B_N" | "C" | "C_N" => Direction::I,
151                "Y" => Direction::O,
152                _ => unknown_pin(macro_name, pin_name),
153            },
154
155            // 4-input NOR: A, B, C, D -> Y (nor4b/nor4bb have _N variants)
156            t if t.starts_with("nor4") => match pin {
157                "A" | "A_N" | "B" | "B_N" | "C" | "C_N" | "D" | "D_N" => Direction::I,
158                "Y" => Direction::O,
159                _ => unknown_pin(macro_name, pin_name),
160            },
161
162            // Majority gate: A, B, C -> X
163            t if t.starts_with("maj3") => match pin {
164                "A" | "B" | "C" => Direction::I,
165                "X" => Direction::O,
166                _ => unknown_pin(macro_name, pin_name),
167            },
168
169            // 2-input XOR: A, B -> X
170            t if t.starts_with("xor2") => match pin {
171                "A" | "B" => Direction::I,
172                "X" => Direction::O,
173                _ => unknown_pin(macro_name, pin_name),
174            },
175
176            // 3-input XOR: A, B, C -> X
177            t if t.starts_with("xor3") => match pin {
178                "A" | "B" | "C" => Direction::I,
179                "X" => Direction::O,
180                _ => unknown_pin(macro_name, pin_name),
181            },
182
183            // 2-input XNOR: A, B -> Y
184            t if t.starts_with("xnor2") => match pin {
185                "A" | "B" => Direction::I,
186                "Y" => Direction::O,
187                _ => unknown_pin(macro_name, pin_name),
188            },
189
190            // 3-input XNOR: A, B, C -> X
191            t if t.starts_with("xnor3") => match pin {
192                "A" | "B" | "C" => Direction::I,
193                "X" => Direction::O,
194                _ => unknown_pin(macro_name, pin_name),
195            },
196
197            // MUX2 (mux2, mux2i): A0, A1, S -> X or Y
198            t if t.starts_with("mux2") => match pin {
199                "A0" | "A1" | "S" => Direction::I,
200                "X" | "Y" => Direction::O,
201                _ => unknown_pin(macro_name, pin_name),
202            },
203
204            // MUX4: A0, A1, A2, A3, S0, S1 -> X
205            t if t.starts_with("mux4") => match pin {
206                "A0" | "A1" | "A2" | "A3" | "S0" | "S1" => Direction::I,
207                "X" => Direction::O,
208                _ => unknown_pin(macro_name, pin_name),
209            },
210
211            // D flip-flop (dfxtp): CLK, D -> Q
212            t if t == "dfxtp" => match pin {
213                "CLK" | "D" => Direction::I,
214                "Q" => Direction::O,
215                _ => unknown_pin(macro_name, pin_name),
216            },
217
218            // D flip-flop with reset (dfrtp, dfrbp): CLK, D, RESET_B -> Q
219            t if t == "dfrtp" || t == "dfrbp" => match pin {
220                "CLK" | "D" | "RESET_B" => Direction::I,
221                "Q" => Direction::O,
222                _ => unknown_pin(macro_name, pin_name),
223            },
224
225            // D flip-flop with set (dfstp): CLK, D, SET_B -> Q
226            t if t == "dfstp" => match pin {
227                "CLK" | "D" | "SET_B" => Direction::I,
228                "Q" => Direction::O,
229                _ => unknown_pin(macro_name, pin_name),
230            },
231
232            // D flip-flop with set and reset (dfbbp): CLK, D, RESET_B, SET_B -> Q, Q_N
233            t if t == "dfbbp" => match pin {
234                "CLK" | "D" | "RESET_B" | "SET_B" => Direction::I,
235                "Q" | "Q_N" => Direction::O,
236                _ => unknown_pin(macro_name, pin_name),
237            },
238
239            // D flip-flop with enable (edfxtp): CLK, D, DE -> Q
240            t if t == "edfxtp" => match pin {
241                "CLK" | "D" | "DE" => Direction::I,
242                "Q" => Direction::O,
243                _ => unknown_pin(macro_name, pin_name),
244            },
245
246            // Scan D flip-flop (sdfxtp): CLK, D, SCD, SCE -> Q
247            t if t == "sdfxtp" => match pin {
248                "CLK" | "D" | "SCD" | "SCE" => Direction::I,
249                "Q" => Direction::O,
250                _ => unknown_pin(macro_name, pin_name),
251            },
252
253            // Latch (dlxtp): D, GATE -> Q
254            t if t.starts_with("dlxtp") || t.starts_with("dlat") => match pin {
255                "D" | "GATE" => Direction::I,
256                "Q" => Direction::O,
257                _ => unknown_pin(macro_name, pin_name),
258            },
259
260            // Tie cells (conb): -> HI, LO
261            t if t == "conb" => match pin {
262                "HI" | "LO" => Direction::O,
263                _ => unknown_pin(macro_name, pin_name),
264            },
265
266            // Half adder: A, B -> COUT, SUM
267            t if t == "ha" => match pin {
268                "A" | "B" => Direction::I,
269                "COUT" | "SUM" => Direction::O,
270                _ => unknown_pin(macro_name, pin_name),
271            },
272
273            // Full adder: A, B, CIN -> COUT, SUM
274            t if t == "fa" => match pin {
275                "A" | "B" | "CIN" => Direction::I,
276                "COUT" | "SUM" => Direction::O,
277                _ => unknown_pin(macro_name, pin_name),
278            },
279
280            // AOI cells - ORDER MATTERS: more specific patterns first!
281
282            // a2111oi: Y = !((A1 & A2) | B1 | C1 | D1)
283            t if t.starts_with("a2111") => match pin {
284                "A1" | "A2" | "B1" | "C1" | "D1" => Direction::I,
285                "X" | "Y" => Direction::O,
286                _ => unknown_pin(macro_name, pin_name),
287            },
288
289            // a211oi: Y = !((A1 & A2) | B1 | C1)
290            t if t.starts_with("a211") => match pin {
291                "A1" | "A2" | "B1" | "C1" => Direction::I,
292                "X" | "Y" => Direction::O,
293                _ => unknown_pin(macro_name, pin_name),
294            },
295
296            // a221oi: Y = !((A1 & A2) | (B1 & B2) | C1)
297            t if t.starts_with("a221") => match pin {
298                "A1" | "A2" | "B1" | "B2" | "C1" => Direction::I,
299                "X" | "Y" => Direction::O,
300                _ => unknown_pin(macro_name, pin_name),
301            },
302
303            // a222oi: Y = !((A1 & A2) | (B1 & B2) | (C1 & C2))
304            t if t.starts_with("a222") => match pin {
305                "A1" | "A2" | "B1" | "B2" | "C1" | "C2" => Direction::I,
306                "X" | "Y" => Direction::O,
307                _ => unknown_pin(macro_name, pin_name),
308            },
309
310            // a22oi: Y = !((A1 & A2) | (B1 & B2))
311            t if t.starts_with("a22") => match pin {
312                "A1" | "A2" | "B1" | "B2" => Direction::I,
313                "X" | "Y" => Direction::O,
314                _ => unknown_pin(macro_name, pin_name),
315            },
316
317            // a2bb2oi: Y = !((!A1_N & !A2_N) | (B1 & B2))
318            t if t.starts_with("a2bb2") => match pin {
319                "A1_N" | "A2_N" | "B1" | "B2" => Direction::I,
320                "X" | "Y" => Direction::O,
321                _ => unknown_pin(macro_name, pin_name),
322            },
323
324            // a21oi Y = !((A1 & A2) | B1)
325            // a21o  Y = (A1 & A2) | B1
326            // a21boi: Y = !((A1 & A2) | !B1_N) = !((A1 & A2) | B1) with B1 pre-inverted
327            t if t.starts_with("a21") => match pin {
328                "A1" | "A2" | "B1" | "B1_N" => Direction::I,
329                "X" | "Y" => Direction::O,
330                _ => unknown_pin(macro_name, pin_name),
331            },
332
333            // a311oi: Y = !((A1 & A2 & A3) | B1 | C1)
334            t if t.starts_with("a311") => match pin {
335                "A1" | "A2" | "A3" | "B1" | "C1" => Direction::I,
336                "X" | "Y" => Direction::O,
337                _ => unknown_pin(macro_name, pin_name),
338            },
339
340            // a31oi: Y = !((A1 & A2 & A3) | B1)
341            t if t.starts_with("a31") => match pin {
342                "A1" | "A2" | "A3" | "B1" => Direction::I,
343                "X" | "Y" => Direction::O,
344                _ => unknown_pin(macro_name, pin_name),
345            },
346
347            // a32oi: Y = !((A1 & A2 & A3) | (B1 & B2))
348            t if t.starts_with("a32") => match pin {
349                "A1" | "A2" | "A3" | "B1" | "B2" => Direction::I,
350                "X" | "Y" => Direction::O,
351                _ => unknown_pin(macro_name, pin_name),
352            },
353
354            // a41oi: Y = !((A1 & A2 & A3 & A4) | B1)
355            t if t.starts_with("a41") => match pin {
356                "A1" | "A2" | "A3" | "A4" | "B1" => Direction::I,
357                "X" | "Y" => Direction::O,
358                _ => unknown_pin(macro_name, pin_name),
359            },
360
361            // OAI cells - ORDER MATTERS: more specific patterns first!
362
363            // o2111ai: Y = !((A1 | A2) & B1 & C1 & D1)
364            t if t.starts_with("o2111") => match pin {
365                "A1" | "A2" | "B1" | "C1" | "D1" => Direction::I,
366                "X" | "Y" => Direction::O,
367                _ => unknown_pin(macro_name, pin_name),
368            },
369
370            // o211ai: Y = !((A1 | A2) & B1 & C1)
371            t if t.starts_with("o211") => match pin {
372                "A1" | "A2" | "B1" | "C1" => Direction::I,
373                "X" | "Y" => Direction::O,
374                _ => unknown_pin(macro_name, pin_name),
375            },
376
377            // o221ai: Y = !((A1 | A2) & (B1 | B2) & C1)
378            t if t.starts_with("o221") => match pin {
379                "A1" | "A2" | "B1" | "B2" | "C1" => Direction::I,
380                "X" | "Y" => Direction::O,
381                _ => unknown_pin(macro_name, pin_name),
382            },
383
384            // o22ai: Y = !((A1 | A2) & (B1 | B2))
385            t if t.starts_with("o22") => match pin {
386                "A1" | "A2" | "B1" | "B2" => Direction::I,
387                "X" | "Y" => Direction::O,
388                _ => unknown_pin(macro_name, pin_name),
389            },
390
391            // o2bb2ai: Y = !((!A1_N | !A2_N) & (B1 | B2))
392            t if t.starts_with("o2bb2") => match pin {
393                "A1_N" | "A2_N" | "B1" | "B2" => Direction::I,
394                "X" | "Y" => Direction::O,
395                _ => unknown_pin(macro_name, pin_name),
396            },
397
398            // o21ai Y = !((A1 | A2) & B1)
399            // o21a  Y = (A1 | A2) & B1
400            // o21bai: Y = !((A1 | A2) & !B1_N)
401            t if t.starts_with("o21") => match pin {
402                "A1" | "A2" | "B1" | "B1_N" => Direction::I,
403                "X" | "Y" => Direction::O,
404                _ => unknown_pin(macro_name, pin_name),
405            },
406
407            // o311ai: Y = !((A1 | A2 | A3) & B1 & C1)
408            t if t.starts_with("o311") => match pin {
409                "A1" | "A2" | "A3" | "B1" | "C1" => Direction::I,
410                "X" | "Y" => Direction::O,
411                _ => unknown_pin(macro_name, pin_name),
412            },
413
414            // o31ai: Y = !((A1 | A2 | A3) & B1)
415            t if t.starts_with("o31") => match pin {
416                "A1" | "A2" | "A3" | "B1" => Direction::I,
417                "X" | "Y" => Direction::O,
418                _ => unknown_pin(macro_name, pin_name),
419            },
420
421            // o32ai: Y = !((A1 | A2 | A3) & (B1 | B2))
422            t if t.starts_with("o32") => match pin {
423                "A1" | "A2" | "A3" | "B1" | "B2" => Direction::I,
424                "X" | "Y" => Direction::O,
425                _ => unknown_pin(macro_name, pin_name),
426            },
427
428            // o41ai: Y = !((A1 | A2 | A3 | A4) & B1)
429            t if t.starts_with("o41") => match pin {
430                "A1" | "A2" | "A3" | "A4" | "B1" => Direction::I,
431                "X" | "Y" => Direction::O,
432                _ => unknown_pin(macro_name, pin_name),
433            },
434
435            // Complex BB cells (two inverted inputs)
436            // a2bb2oi: Y = !((!A1_N & !A2_N) | (B1 & B2)) = !((A1_N | A2_N) | (B1 & B2))
437            t if t.starts_with("a2bb2o") => match pin {
438                "A1_N" | "A2_N" | "B1" | "B2" => Direction::I,
439                "X" | "Y" => Direction::O,
440                _ => unknown_pin(macro_name, pin_name),
441            },
442
443            // o2bb2ai: Y = !((!A1_N | !A2_N) & (B1 | B2)) = !((A1_N & A2_N) & (B1 | B2))
444            t if t.starts_with("o2bb2a") => match pin {
445                "A1_N" | "A2_N" | "B1" | "B2" => Direction::I,
446                "X" | "Y" => Direction::O,
447                _ => unknown_pin(macro_name, pin_name),
448            },
449
450            // Filler/tap/decap cells - no functional pins
451            t if t.starts_with("fill") || t.starts_with("tap") || t.starts_with("decap") => {
452                // These cells have no functional pins, but may have power pins
453                // which we treat as inputs (though they're typically not connected in logic)
454                Direction::I
455            }
456
457            _ => {
458                // Check for ChipFlow SRAM macros (CF_SRAM_*)
459                if macro_name.starts_with("CF_SRAM_") {
460                    match pin {
461                        // Control inputs
462                        "WLBI" | "WLOFF" | "CLKin" | "EN" | "R_WB" | "SM" | "TM" => Direction::I,
463                        "ScanInDR" | "ScanInDL" | "ScanInCC" => Direction::I,
464                        "vpwrpc" | "vpwrac" => Direction::I,
465                        // Data/address inputs
466                        "AD" | "BEN" | "DI" => Direction::I,
467                        // Data/scan outputs
468                        "DO" | "ScanOutCC" => Direction::O,
469                        _ => unknown_pin(macro_name, pin_name),
470                    }
471                } else {
472                    unknown_pin(macro_name, pin_name)
473                }
474            }
475        }
476    }
477
478    fn width_of(
479        &self,
480        macro_name: &CompactString,
481        pin_name: &CompactString,
482    ) -> Option<SVerilogRange> {
483        // Check for ChipFlow SRAM macros which have bus pins
484        if macro_name.starts_with("CF_SRAM_") {
485            match pin_name.as_str() {
486                // Data buses are 32-bit
487                "DO" | "DI" | "BEN" => Some(SVerilogRange(31, 0)),
488                // Address bus - width depends on SRAM depth
489                // CF_SRAM_1024x32 has 10-bit address (2^10 = 1024)
490                "AD" => Some(SVerilogRange(9, 0)),
491                // All other pins are single-bit
492                _ => None,
493            }
494        } else {
495            // All SKY130 standard cells have single-bit pins
496            None
497        }
498    }
499}

501/// Extract the cell type from a SKY130 macro name.
502///
503/// Converts `sky130_fd_sc_hd__inv_2` to `inv`.
504pub fn extract_cell_type(name: &str) -> &str {
505    // First strip the library prefix
506    let stripped = name
507        .strip_prefix("sky130_fd_sc_hd__")
508        .or_else(|| name.strip_prefix("sky130_fd_sc_hs__"))
509        .or_else(|| name.strip_prefix("sky130_fd_sc_ms__"))
510        .or_else(|| name.strip_prefix("sky130_fd_sc_ls__"))
511        .or_else(|| name.strip_prefix("sky130_fd_sc_lp__"))
512        .or_else(|| name.strip_prefix("sky130_fd_sc_hdll__"))
513        .or_else(|| name.strip_prefix("sky130_fd_sc_hvl__"))
514        .unwrap_or(name);
515
516    // Then strip the drive strength suffix (e.g., _2, _4, _16)
517    // The drive strength is always a number at the end after an underscore
518    if let Some(last_underscore) = stripped.rfind('_') {
519        let suffix = &stripped[last_underscore + 1..];
520        // Check if suffix is purely numeric (drive strength)
521        if !suffix.is_empty() && suffix.chars().all(|c| c.is_ascii_digit()) {
522            return &stripped[..last_underscore];
523        }
524    }
525
526    stripped
527}

529/// Return Direction::I as fallback but log an error.
530fn unknown_pin(macro_name: &CompactString, pin_name: &CompactString) -> Direction {
531    panic!("Unknown SKY130 pin: macro={}, pin={}", macro_name, pin_name);
532}

534/// Check if a cell name is a SKY130 standard cell or associated macro.
535pub fn is_sky130_cell(name: &str) -> bool {
536    name.starts_with("sky130_fd_sc_") || name.starts_with("CF_SRAM_")
537}
538
539/// Check if a cell name is an AIGPDK cell.
540pub fn is_aigpdk_cell(name: &str) -> bool {
541    matches!(
542        name,
543        "INV"
544            | "BUF"
545            | "CKLNQD"
546            | "AND2_00_0"
547            | "AND2_01_0"
548            | "AND2_10_0"
549            | "AND2_11_0"
550            | "AND2_11_1"
551            | "DFF"
552            | "DFFSR"
553            | "LATCH"
554            | "GEM_ASSERT"
555            | "GEM_DISPLAY"
556    ) || name.starts_with("$__RAMGEM_")
557}

559/// The detected cell library type.
560#[derive(Debug, Clone, Copy, PartialEq, Eq)]
561pub enum CellLibrary {
562    /// AIGPDK standard cells
563    AIGPDK,
564    /// SKY130 open-source PDK
565    SKY130,
566    /// Mixed cells from both libraries (error case)
567    Mixed,
568}
569
570impl std::fmt::Display for CellLibrary {
571    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
572        match self {
573            CellLibrary::AIGPDK => write!(f, "AIGPDK"),
574            CellLibrary::SKY130 => write!(f, "SKY130"),
575            CellLibrary::Mixed => write!(f, "Mixed"),
576        }
577    }
578}
579
580/// Detect which cell library is used in a netlist.
581///
582/// Scans all cells in the netlist and returns:
583/// - `AIGPDK` if only AIGPDK cells are found
584/// - `SKY130` if only SKY130 cells are found
585/// - `Mixed` if both types are found (this is an error condition)
586///
587/// If no recognizable cells are found, defaults to AIGPDK.
588pub fn detect_library(cell_types: impl Iterator<Item = impl AsRef<str>>) -> CellLibrary {
589    let mut has_aigpdk = false;
590    let mut has_sky130 = false;
591
592    for cell_type in cell_types {
593        let name = cell_type.as_ref();
594        if is_sky130_cell(name) {
595            has_sky130 = true;
596        } else if is_aigpdk_cell(name) {
597            has_aigpdk = true;
598        }
599        // Early exit if we've found both
600        if has_aigpdk && has_sky130 {
601            return CellLibrary::Mixed;
602        }
603    }
604
605    match (has_aigpdk, has_sky130) {
606        (_, true) => CellLibrary::SKY130,
607        (true, false) => CellLibrary::AIGPDK,
608        (false, false) => CellLibrary::AIGPDK, // Default
609    }
610}

612/// Detect library from a Verilog file by scanning for cell types.
613///
614/// This does a quick scan of the file without full parsing.
615pub fn detect_library_from_file(path: &std::path::Path) -> std::io::Result<CellLibrary> {
616    use std::fs::File;
617    use std::io::{BufRead, BufReader};
618
619    let file = File::open(path)?;
620    let reader = BufReader::new(file);
621
622    let mut has_aigpdk = false;
623    let mut has_sky130 = false;
624
625    for line in reader.lines() {
626        let line = line?;
627        // Quick pattern matching for cell instantiations
628        if line.contains("sky130_fd_sc_") {
629            has_sky130 = true;
630        }
631        // Look for AIGPDK cells (typically at start of instantiation)
632        for cell in [
633            "AND2_",
634            "INV ",
635            "BUF ",
636            "DFF ",
637            "DFFSR ",
638            "LATCH ",
639            "CKLNQD ",
640            "GEM_ASSERT ",
641            "GEM_DISPLAY ",
642            "$__RAMGEM_",
643        ] {
644            if line.contains(cell) {
645                has_aigpdk = true;
646                break;
647            }
648        }
649        // Early exit if mixed
650        if has_aigpdk && has_sky130 {
651            return Ok(CellLibrary::Mixed);
652        }
653    }
654
655    Ok(match (has_aigpdk, has_sky130) {
656        (_, true) => CellLibrary::SKY130,
657        (true, false) => CellLibrary::AIGPDK,
658        (false, false) => CellLibrary::AIGPDK,
659    })
660}
