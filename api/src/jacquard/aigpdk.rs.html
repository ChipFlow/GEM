3//! AIGPDK is a special artificial cell library used in GEM.
4
5use compact_str::CompactString;
6use netlistdb::{Direction, LeafPinProvider};
7use sverilogparse::SVerilogRange;
8
9/// This implements direction and width providers for
10/// AIG PDK cells.
11///
12/// You can use it in netlistdb construction.
13pub struct AIGPDKLeafPins();
14
15/// The addr width of an SRAM.
16///
17/// The word width is always 32.
18/// If you change this, make sure to change all other occurences in this
19/// project as well as the definitions in PDK libraries.
20pub const AIGPDK_SRAM_ADDR_WIDTH: usize = 13;
21
22pub const AIGPDK_SRAM_SIZE: usize = 1 << 13;
23
24impl LeafPinProvider for AIGPDKLeafPins {
25    fn direction_of(
26        &self,
27        macro_name: &CompactString,
28        pin_name: &CompactString,
29        pin_idx: Option<isize>,
30    ) -> Direction {
31        match (macro_name.as_str(), pin_name.as_str(), pin_idx) {
32            ("INV" | "BUF", "A", None) => Direction::I,
33            ("INV" | "BUF", "Y", None) => Direction::O,
34
35            (
36                "AND2_00_0" | "AND2_01_0" | "AND2_10_0" | "AND2_11_0" | "AND2_11_1",
37                "A" | "B",
38                None,
39            ) => Direction::I,
40            ("AND2_00_0" | "AND2_01_0" | "AND2_10_0" | "AND2_11_0" | "AND2_11_1", "Y", None) => {
41                Direction::O
42            }
43
44            ("DFF" | "LATCH", "CLK" | "D", None) => Direction::I,
45            ("DFFSR", "CLK" | "D" | "S" | "R", None) => Direction::I,
46            ("DFF" | "DFFSR" | "LATCH", "Q", None) => Direction::O,
47
48            ("CKLNQD", "CP" | "E", None) => Direction::I,
49            ("CKLNQD", "Q", None) => Direction::O,
50
51            // GEM assertion and display cells (no outputs)
52            ("GEM_ASSERT", "CLK" | "EN" | "A", None) => Direction::I,
53            ("GEM_DISPLAY", "CLK" | "EN", None) => Direction::I,
54            ("GEM_DISPLAY", "MSG_ID", Some(0..=31)) => Direction::I,
55
56            ("$__RAMGEM_ASYNC_", _, _) => {
57                panic!(
58                    "Async RAM (lib cell {}) not supported yet in GEM.",
59                    macro_name
60                );
61            }
62
63            ("$__RAMGEM_SYNC_", "PORT_R_CLK" | "PORT_W_CLK", None) => Direction::I,
64            ("$__RAMGEM_SYNC_", "PORT_R_ADDR" | "PORT_W_ADDR", Some(0..=12)) => Direction::I,
65            ("$__RAMGEM_SYNC_", "PORT_W_WR_EN" | "PORT_W_WR_DATA", Some(0..=31)) => Direction::I,
66            ("$__RAMGEM_SYNC_", "PORT_R_RD_DATA", Some(0..=31)) => Direction::O,
67
68            _ => {
69                use netlistdb::{GeneralPinName, HierName};
70                panic!("Cannot recognize pin type {}, please make sure the verilog netlist is synthesized in GEM's aigpdk.",
71                       (HierName::single(macro_name.clone()),
72                        pin_name, pin_idx).dbg_fmt_pin());
73            }
74        }
75    }
76
77    fn width_of(
78        &self,
79        macro_name: &CompactString,
80        pin_name: &CompactString,
81    ) -> Option<SVerilogRange> {
82        match (macro_name.as_str(), pin_name.as_str()) {
83            ("INV" | "BUF", "A" | "Y") => None,
84            (
85                "AND2_00_0" | "AND2_01_0" | "AND2_10_0" | "AND2_11_0" | "AND2_11_1",
86                "A" | "B" | "Y",
87            ) => None,
88            ("DFF" | "DFFSR" | "LATCH", "CLK" | "D" | "Q" | "S" | "R") => None,
89            ("CKLNQD", "CP" | "E" | "Q") => None,
90            // GEM assertion and display cells
91            ("GEM_ASSERT", "CLK" | "EN" | "A") => None,
92            ("GEM_DISPLAY", "CLK" | "EN") => None,
93            ("GEM_DISPLAY", "MSG_ID") => Some(SVerilogRange(31, 0)),
94            ("$__RAMGEM_SYNC_", "PORT_R_CLK" | "PORT_W_CLK") => None,
95            ("$__RAMGEM_SYNC_", "PORT_R_ADDR" | "PORT_W_ADDR") => Some(SVerilogRange(12, 0)),
96            ("$__RAMGEM_SYNC_", "PORT_W_WR_EN" | "PORT_W_WR_DATA" | "PORT_R_RD_DATA") => {
97                Some(SVerilogRange(31, 0))
98            }
99            _ => None,
100        }
101    }
102}
