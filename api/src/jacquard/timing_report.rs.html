1//! End-of-run structured timing report (`--timing-report <path.json>`).
2//!
3//! Schema versioning policy mirrors the timing IR (ADR 0002): explicit
4//! `schema_version` field, additive-only extension, breaking changes
5//! require a major bump and migration notes (per ADR 0008's stability
6//! contract).
7//!
8//! `worst_slack` is currently populated only from violation events.
9//! Closest-to-violation tracking when no violation ever occurred needs
10//! GPU-side near-miss instrumentation and is not in this module's scope.
11

13use std::path::Path;

17/// Schema version for the timing report JSON. Bump the major component
18/// for breaking changes; minor for additive fields per the ADR 0008
19/// stability contract.
20pub const SCHEMA_VERSION: &str = "1.0.0";

22/// One setup or hold violation record. Mirrors the `EventType::*Violation`
23/// payload, with `word_id` resolved to a human-readable `site` string via
24/// the caller's `WordSymbolMap`-backed resolver.
25#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
26pub struct ViolationRecord {
27    pub cycle: u32,
28    pub kind: ViolationKind,
29    pub word_id: u32,
30    pub site: String,
31    pub arrival_ps: u32,
32    pub constraint_ps: u32,
33    pub slack_ps: i32,
34}
35
36#[derive(Debug, Clone, Copy, Serialize, Deserialize, PartialEq, Eq)]
37#[serde(rename_all = "snake_case")]
38pub enum ViolationKind {
39    Setup,
40    Hold,
41}

43/// Run-time metadata. All fields optional except those Jacquard always
44/// knows (clock period, version).
45#[derive(Debug, Clone, Default, Serialize, Deserialize)]
46pub struct RunMetadata {
47    pub design: Option<String>,
48    pub vector_source: Option<String>,
49    pub timing_source: Option<String>,
50    pub clock_period_ps: u64,
51    pub cycles_run: u32,
52    pub jacquard_version: String,
53}

55/// Aggregate counters mirroring `SimStats` violation-related fields.
56#[derive(Debug, Clone, Default, Serialize, Deserialize)]
57pub struct ReportStats {

59    pub hold_violations: u32,

63/// Per-word aggregate: violation counts and worst slack across the run.
64/// `site` matches the formatted descriptor `WordSymbolMap` produces for
65/// `word_id`. `Option`s distinguish "no violation of this kind seen"
66/// from a literal-zero observation.
67#[derive(Debug, Clone, Serialize, Deserialize, PartialEq, Eq)]
68pub struct PerWordSummary {
69    pub word_id: u32,
70    pub site: String,
71    pub setup_violations: u32,
72    pub hold_violations: u32,
73    pub worst_setup_slack_ps: Option<i32>,
74    pub worst_hold_slack_ps: Option<i32>,
75    pub worst_arrival_ps: Option<u32>,
76}

78/// Top-N worst-slack rankings per kind.
79#[derive(Debug, Clone, Default, Serialize, Deserialize)]
80pub struct WorstSlack {

85/// Top-level report document.
86#[derive(Debug, Clone, Serialize, Deserialize)]
87pub struct TimingReport {
88    pub schema_version: String,
89    pub metadata: RunMetadata,
90    pub stats: ReportStats,
91    pub violations: Vec<ViolationRecord>,
92    pub per_word: Vec<PerWordSummary>,
93    pub worst_slack: WorstSlack,
94}

96impl TimingReport {
97    pub fn new(metadata: RunMetadata) -> Self {
98        Self {
99            schema_version: SCHEMA_VERSION.to_string(),
100            metadata,
101            stats: ReportStats::default(),
102            violations: Vec::new(),
103            per_word: Vec::new(),
104            worst_slack: WorstSlack::default(),
105        }
106    }
107
108    /// Serialize as pretty JSON to the given writer.
109    pub fn write_pretty<W: std::io::Write>(&self, w: &mut W) -> serde_json::Result<()> {
110        serde_json::to_writer_pretty(w, self)
111    }
112
113    /// Convenience wrapper: write the report to a file path. Creates
114    /// parent directories if needed.
115    pub fn write_to_path(&self, path: &Path) -> std::io::Result<()> {
116        if let Some(parent) = path.parent() {
117            if !parent.as_os_str().is_empty() {
118                std::fs::create_dir_all(parent)?;
119            }
120        }
121        let f = std::fs::File::create(path)?;
122        let mut w = std::io::BufWriter::new(f);
123        self.write_pretty(&mut w)
124            .map_err(|e| std::io::Error::other(e.to_string()))?;
125        Ok(())
126    }

128    /// Render the report as a multi-line text summary suitable for
129    /// stdout. Matches `--timing-summary`. Stable enough for human
130    /// inspection but explicitly NOT a parseable contract per ADR 0008
131    /// (`--timing-report` JSON is the parseable form).
132    ///
133    /// ADR 0008 also lists "corner" and "margin percentage" as desired
134    /// summary content; both are deferred — corner because the
135    /// metadata struct doesn't carry it yet (next IR roundtrip), margin
136    /// because it duplicates information the consumer can compute from
137    /// `slack_ps` and `clock_period_ps` already in the report.
138    pub fn format_summary(&self) -> String {
139        // Top-N for the per-word count breakdown. Distinct from
140        // jacquard.rs::WORST_SLACK_TOP_N (which gates the JSON ranking
141        // capacity); this one bounds the human-facing table width.
142        const SUMMARY_TOP_N: usize = 5;
143
144        use std::fmt::Write as _;
145        let mut out = String::new();
146        // `writeln!` into a String via fmt::Write is infallible.
147        writeln!(out, "=== Jacquard Timing Summary ===").unwrap();
148        if let Some(d) = &self.metadata.design {
149            writeln!(out, "Design:        {d}").unwrap();
150        }
151        match &self.metadata.vector_source {
152            Some(v) => writeln!(
153                out,
154                "Vectors:       {v} ({} cycles)",
155                self.metadata.cycles_run
156            )
157            .unwrap(),
158            None => writeln!(out, "Vectors:       {} cycles", self.metadata.cycles_run).unwrap(),
159        }
160        writeln!(out, "Clock period:  {} ps", self.metadata.clock_period_ps).unwrap();
161        if let Some(t) = &self.metadata.timing_source {
162            writeln!(out, "Timing source: {t}").unwrap();
163        }
164        writeln!(out).unwrap();
165
166        writeln!(out, "Violations:").unwrap();
167        writeln!(out, "  Setup: {}", self.stats.setup_violations).unwrap();
168        writeln!(out, "  Hold:  {}", self.stats.hold_violations).unwrap();
169        let total = self.stats.setup_violations + self.stats.hold_violations;
170        writeln!(out, "  Total: {total}").unwrap();
171        if self.stats.events_dropped > 0 {
172            writeln!(
173                out,
174                "  Note:  {} events dropped (buffer overflow)",
175                self.stats.events_dropped
176            )
177            .unwrap();
178        }
179        writeln!(out).unwrap();
180
181        writeln!(out, "Worst slack:").unwrap();
182        write_worst(&mut out, "Setup", self.worst_slack.setup.first());
183        write_worst(&mut out, "Hold", self.worst_slack.hold.first());
184
185        if !self.per_word.is_empty() {
186            writeln!(out).unwrap();
187            let shown = self.per_word.len().min(SUMMARY_TOP_N);
188            writeln!(
189                out,
190                "Top {shown} by violation count (of {} total words with violations):",
191                self.per_word.len()
192            )
193            .unwrap();
194            for w in self.per_word.iter().take(shown) {
195                let total = w.setup_violations + w.hold_violations;
196                writeln!(
197                    out,
198                    "  {} ({} violations): worst setup={} hold={} arrival={}",
199                    w.site,
200                    total,
201                    fmt_optional_ps(w.worst_setup_slack_ps),
202                    fmt_optional_ps(w.worst_hold_slack_ps),
203                    fmt_optional_ps(w.worst_arrival_ps.map(|a| a as i64)),
204                )
205                .unwrap();
206            }
207        }
208
209        out
210    }
211}

213fn write_worst(out: &mut String, label: &str, v: Option<&ViolationRecord>) {
214    use std::fmt::Write as _;
215    match v {
216        Some(v) => writeln!(
217            out,
218            "  {label}: {}ps  at {}  (cycle {})",
219            v.slack_ps, v.site, v.cycle
220        )
221        .unwrap(),
222        None => writeln!(out, "  {label}: (none)").unwrap(),
223    }
224}

226/// Render an `Option<i64>` (representing picoseconds) as `"-"` for
227/// `None` and `"<n>ps"` for `Some`. Generic over the integer side via
228/// the `Into<i64>` callsite cast above.
229fn fmt_optional_ps(v: Option<impl Into<i64>>) -> String {
230    match v {
231        Some(n) => format!("{}ps", n.into()),
232        None => "-".into(),
233    }
234}

236/// Accumulator that observes every violation as it occurs, tracking the
237/// running per-word aggregates and the top-N worst-slack rankings, and
238/// keeping the full per-cycle list. Materialised into a `TimingReport`
239/// at end of run via `finalize`.
240pub struct ReportBuilder {
241    metadata: RunMetadata,
242    violations: Vec<ViolationRecord>,
243    setup_worst: WorstSlackTracker,
244    hold_worst: WorstSlackTracker,
245}
246
247impl ReportBuilder {
248    pub fn new(metadata: RunMetadata, worst_slack_n: usize) -> Self {
249        Self {
250            metadata,
251            violations: Vec::new(),
252            setup_worst: WorstSlackTracker::with_capacity(worst_slack_n),
253            hold_worst: WorstSlackTracker::with_capacity(worst_slack_n),
254        }
255    }
256
257    pub fn observe(&mut self, v: ViolationRecord) {
258        match v.kind {
259            ViolationKind::Setup => self.setup_worst.push(v.clone()),
260            ViolationKind::Hold => self.hold_worst.push(v.clone()),
261        }
262        self.violations.push(v);
263    }
264
265    /// Finalise into a `TimingReport`. `cycles_run` and the aggregate
266    /// stats are passed in (they live outside the builder); `events_dropped`
267    /// is part of `stats`.
268    pub fn finalize(mut self, cycles_run: u32, stats: ReportStats) -> TimingReport {
269        self.metadata.cycles_run = cycles_run;
270        let per_word = aggregate_per_word(&self.violations);
271        TimingReport {
272            schema_version: SCHEMA_VERSION.to_string(),
273            metadata: self.metadata,
274            stats,
275            violations: self.violations,
276            per_word,
277            worst_slack: WorstSlack {
278                setup: self.setup_worst.into_sorted_vec(),
279                hold: self.hold_worst.into_sorted_vec(),
280            },
281        }
282    }
283}

285/// Bounded top-N tracker keeping the most-negative slacks (closest to
286/// or beyond violation). Internally a `Vec` since N is small (default 10).
287struct WorstSlackTracker {
288    capacity: usize,
289    items: Vec<ViolationRecord>,
290}
291
292impl WorstSlackTracker {
293    fn with_capacity(capacity: usize) -> Self {
294        Self {
295            capacity: capacity.max(1),
296            items: Vec::with_capacity(capacity.max(1)),
297        }
298    }
299
300    fn push(&mut self, v: ViolationRecord) {
301        // Worst = most negative slack; we keep the smallest N.
302        if self.items.len() < self.capacity {
303            self.items.push(v);
304            return;
305        }
306        // Find the largest (least negative) currently in the set; replace if v is smaller.
307        let (idx, max) = self
308            .items
309            .iter()
310            .enumerate()
311            .max_by(|(_, a), (_, b)| a.slack_ps.cmp(&b.slack_ps))
312            .expect("non-empty by construction");
313        if v.slack_ps < max.slack_ps {
314            self.items[idx] = v;
315        }
316    }
317
318    fn into_sorted_vec(mut self) -> Vec<ViolationRecord> {
319        self.items.sort_by(|a, b| match a.slack_ps.cmp(&b.slack_ps) {
320            Ordering::Equal => a.cycle.cmp(&b.cycle),
321            other => other,
322        });
323        self.items
324    }
325}

327fn aggregate_per_word(violations: &[ViolationRecord]) -> Vec<PerWordSummary> {
328    use std::collections::HashMap;
329    let mut by_word: HashMap<u32, PerWordSummary> = HashMap::new();
330    for v in violations {
331        let entry = by_word
332            .entry(v.word_id)
333            .or_insert_with(|| PerWordSummary {
334                word_id: v.word_id,
335                site: v.site.clone(),
336                setup_violations: 0,
337                hold_violations: 0,
338                worst_setup_slack_ps: None,
339                worst_hold_slack_ps: None,
340                worst_arrival_ps: None,
341            });
342        match v.kind {
343            ViolationKind::Setup => {
344                entry.setup_violations += 1;
345                entry.worst_setup_slack_ps = Some(match entry.worst_setup_slack_ps {
346                    Some(s) => s.min(v.slack_ps),
347                    None => v.slack_ps,
348                });
349            }
350            ViolationKind::Hold => {
351                entry.hold_violations += 1;
352                entry.worst_hold_slack_ps = Some(match entry.worst_hold_slack_ps {
353                    Some(s) => s.min(v.slack_ps),
354                    None => v.slack_ps,
355                });
356            }
357        }
358        entry.worst_arrival_ps = Some(match entry.worst_arrival_ps {
359            Some(a) => a.max(v.arrival_ps),
360            None => v.arrival_ps,
361        });
362    }
363    let mut out: Vec<_> = by_word.into_values().collect();
364    out.sort_by(|a, b| {
365        let total_a = a.setup_violations + a.hold_violations;
366        let total_b = b.setup_violations + b.hold_violations;
367        total_b.cmp(&total_a).then(a.word_id.cmp(&b.word_id))
368    });
369    out
370}
