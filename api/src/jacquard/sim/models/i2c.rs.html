3//! I²C peripheral model — port of chipflow-lib's
4//! `chipflow/common/sim/models.cc::i2c::step` (lines 314-374).
5//!
6//! Open-drain semantics: the model and design both can pull SDA/SCL
7//! low; high is "released" (the bus is pulled high by an external
8//! pull-up). Bus value = wired-AND of all drivers. The model reads
9//! the design's `sda_oe`/`scl_oe` outputs (oe=1 → design pulling low)
10//! and contributes its own `drive_sda` (false → model pulling low,
11//! true → released).
12//!
13//! State machine: detects START (SDA falls while SCL high), STOP (SDA
14//! rises while SCL high), and per-bit shifts on SCL edges. After each
15//! 8-bit byte, on the next SCL low the model either ACKs/NACKs (set by
16//! `ack`/`nack` action), or shifts out the next bit of `read_data`
17//! (set by `set_data` action) when the master is reading.
18//!
19//! Wiring is TODO — needs port-mapping infrastructure to look up the
20//! design's `i2c_<N>` peripheral port positions in the GPU state buffer
21//! by name (Jacquard's `GpioMapping` currently only supports GPIO-index
22//! lookup for inputs and named lookup for inputs but not outputs).
23//! The model itself is fully tested.
24

29/// Pin positions for one I²C peripheral.
30pub struct I2cPins {
31    /// State position of `sda_oe` (design output: 1 → design pulling SDA low).
32    pub sda_oe_out_pos: u32,
33    /// State position of `scl_oe` (design output: 1 → design pulling SCL low).
34    pub scl_oe_out_pos: u32,
35    /// State position of `sda_i` (design input: what design sees).
36    pub sda_i_in_pos: u32,
37    /// State position of `scl_i` (design input: what design sees).
38    pub scl_i_in_pos: u32,
39}
40
41/// CPU-side state for one I²C peripheral.
42pub struct I2cModel {
43    /// Peripheral name (matches chipflow's `i2c_<id>`).
44    pub name: String,
45    pins: I2cPins,
46    /// Byte index in the current transaction (0 = address, 1+ = data).
47    byte_count: u32,
48    /// Bit index in the current byte (0..7 = data, 8 = ack/nack).
49    bit_count: u32,
50    /// Whether to ACK the current byte (queued via `ack` / `nack` action).
51    do_ack: bool,
52    /// Direction inferred from the address byte's R/W bit. True = master read.
53    is_read: bool,
54    /// Byte the model returns when the master is reading
55    /// (queued via `set_data` action).
56    read_data: u8,
57    /// 8-bit shift register.
58    sr: u8,
59    /// Whether the model releases SDA (true) or pulls it low (false).
60    drive_sda: bool,
61    /// Previous-edge SDA / SCL (post-wired-AND with model contributions
62    /// is irrelevant for edge detection — chipflow uses just the
63    /// design-side `sda` / `scl` derived from oe, which is what we mirror).
64    last_sda: bool,
65    last_scl: bool,
66}
67
68impl I2cModel {
69    /// Build a model. Initial state: SDA/SCL released (high), no bytes
70    /// transferred.
71    pub fn new(name: String, pins: I2cPins) -> Self {
72        Self {
73            name,
74            pins,
75            byte_count: 0,
76            bit_count: 0,
77            do_ack: false,
78            is_read: false,
79            read_data: 0,
80            sr: 0xFF,
81            drive_sda: true,
82            last_sda: true,
83            last_scl: true,
84        }
85    }
86
87    /// State positions this model drives (input side).
88    pub fn driven_positions(&self) -> [u32; 2] {
89        [self.pins.sda_i_in_pos, self.pins.scl_i_in_pos]
90    }
91
92    /// Apply a queued action.
93    pub fn apply_action(&mut self, action: &QueuedAction) {
94        match action.event.as_str() {
95            "ack" => self.do_ack = true,
96            "nack" => self.do_ack = false,
97            "set_data" => {
98                self.read_data = action
99                    .payload
100                    .as_u64()
101                    .filter(|&v| v <= 0xFF)
102                    .unwrap_or_else(|| {
103                        panic!(
104                            "i2c `{}`: `set_data` payload must be u8, got {:?}",
105                            self.name, action.payload
106                        );
107                    }) as u8;
108            }
109            other => {
110                clilog::warn!(
111                    "i2c `{}`: unhandled event `{}` (payload {:?})",
112                    self.name,
113                    other,
114                    action.payload
115                );
116            }
117        }
118    }
119
120    /// Advance the I²C state machine one step. Reads `sda_oe` / `scl_oe`
121    /// from the design's output state, runs the FSM, contributes
122    /// SDA/SCL input drives to `overrides`, and pushes any emitted
123    /// events into `emitted`.
124    pub fn step(
125        &mut self,
126        output_state: &[u32],
127        overrides: &mut ModelOverrides,
128        emitted: &mut Vec<EmittedEvent>,
129        timestamp: u64,
130    ) {
131        let sda_oe = read_bit(output_state, self.pins.sda_oe_out_pos);
132        let scl_oe = read_bit(output_state, self.pins.scl_oe_out_pos);
133        // Open-drain: design pulls low when oe=1, releases (high) when oe=0.
134        let sda = sda_oe == 0;
135        let scl = scl_oe == 0;
136
137        if self.last_scl && self.last_sda && !sda {
138            // START condition (SDA falls while SCL high)
139            emitted.push(EmittedEvent {
140                peripheral: self.name.clone(),
141                event: "start".to_string(),
142                payload: json!(""),
143            });
144            let _ = timestamp;
145            self.sr = 0xFF;
146            self.byte_count = 0;
147            self.bit_count = 0;
148            self.is_read = false;
149            self.drive_sda = true;
150        } else if scl && !self.last_scl {
151            // SCL posedge: shift in (during write) or do nothing (during read of next bytes)
152            if self.byte_count == 0 || !self.is_read {
153                self.sr = (self.sr << 1) | (if sda { 1 } else { 0 });
154            }
155            self.bit_count += 1;
156            if self.bit_count == 8 {
157                if self.byte_count == 0 {
158                    self.is_read = (self.sr & 0x1) != 0;
159                    emitted.push(EmittedEvent {
160                        peripheral: self.name.clone(),
161                        event: "address".to_string(),
162                        payload: json!(self.sr),
163                    });
164                } else if !self.is_read {
165                    emitted.push(EmittedEvent {
166                        peripheral: self.name.clone(),
167                        event: "write".to_string(),
168                        payload: json!(self.sr),
169                    });
170                }
171                self.byte_count += 1;
172            } else if self.bit_count == 9 {
173                self.bit_count = 0;
174            }
175        } else if !scl && self.last_scl {
176            // SCL negedge: release SDA, then drive ack/data/release for next bit
177            self.drive_sda = true;
178            if self.bit_count == 8 {
179                self.drive_sda = !self.do_ack;
180            } else if self.byte_count > 0 && self.is_read {
181                if self.bit_count == 0 {
182                    self.sr = self.read_data;
183                } else {
184                    self.sr <<= 1;
185                }
186                self.drive_sda = ((self.sr >> 7) & 0x1) != 0;
187            }
188        } else if self.last_scl && !self.last_sda && sda {
189            // STOP condition (SDA rises while SCL high)
190            emitted.push(EmittedEvent {
191                peripheral: self.name.clone(),
192                event: "stop".to_string(),
193                payload: json!(""),
194            });
195            self.drive_sda = true;
196        }
197
198        self.last_sda = sda;
199        self.last_scl = scl;
200        // sda_i = wired-AND of design (sda) and model (drive_sda).
201        let sda_i = if sda && self.drive_sda { 1 } else { 0 };
202        let scl_i = if scl { 1 } else { 0 };
203        overrides.insert(self.pins.sda_i_in_pos, sda_i);
204        overrides.insert(self.pins.scl_i_in_pos, scl_i);
205    }
206}

208#[inline]
209fn read_bit(state: &[u32], pos: u32) -> u8 {
210    let word = (pos >> 5) as usize;
211    let bit = pos & 31;
212    if word < state.len() {
213        ((state[word] >> bit) & 1) as u8
214    } else {
215        0
216    }
217}

223    /// In-memory bit-vector for tests. We model `sda_oe` and `scl_oe` at
224    /// known positions and sweep them through transactions.
225    struct BusHarness {
226        state: Vec<u32>,
227    }
228
229    impl BusHarness {
230        fn new(size_words: usize) -> Self {
231            Self {
232                state: vec![0u32; size_words],
233            }
234        }
235        fn set_bit(&mut self, pos: u32, value: u8) {
236            let word = (pos >> 5) as usize;
237            let bit = pos & 31;
238            if word < self.state.len() {
239                if value != 0 {
240                    self.state[word] |= 1u32 << bit;
241                } else {
242                    self.state[word] &= !(1u32 << bit);
243                }
244            }
245        }
246    }

248    fn pins() -> I2cPins {
249        I2cPins {
250            sda_oe_out_pos: 0,
251            scl_oe_out_pos: 1,
252            sda_i_in_pos: 2,
253            scl_i_in_pos: 3,
254        }
255    }

257    /// Run several FSM steps in sequence with given (sda, scl) bus
258    /// values (open-drain logical level: 1 = high/released, 0 = low).
259    /// Each (sda, scl) pair is a single step; the harness translates to
260    /// `sda_oe`/`scl_oe` (oe = inverse of bus level).
261    fn run_steps(
262        model: &mut I2cModel,
263        harness: &mut BusHarness,
264        steps: &[(u8, u8)],
265    ) -> Vec<EmittedEvent> {
266        let mut events = Vec::new();
267        for &(sda_high, scl_high) in steps {
268            harness.set_bit(model.pins.sda_oe_out_pos, if sda_high == 0 { 1 } else { 0 });
269            harness.set_bit(model.pins.scl_oe_out_pos, if scl_high == 0 { 1 } else { 0 });
270            let mut overrides = ModelOverrides::new();
271            model.step(&harness.state, &mut overrides, &mut events, 0);
272        }
273        events
274    }

276    /// Helper: emit one full I²C byte by toggling SCL and SDA bit-by-bit.
277    /// `bits` is MSB-first. Real I²C masters change SDA only while SCL is
278    /// low, so each bit takes three sub-steps: drop SCL (preserving the
279    /// previous SDA), change SDA, raise SCL. Initial state expected:
280    /// SCL=0, SDA=0 (typical post-START).
281    fn write_byte_steps(bits: u8) -> Vec<(u8, u8)> {
282        let mut steps = Vec::new();
283        let mut prev = 0u8;
284        for i in 0..8 {
285            let bit = (bits >> (7 - i)) & 0x1;
286            // Drop SCL with previous SDA (clean negedge with no SDA change).
287            steps.push((prev, 0));
288            // Change SDA while SCL is low.
289            steps.push((bit, 0));
290            // Raise SCL — this is the posedge where the slave samples.
291            steps.push((bit, 1));
292            prev = bit;
293        }
294        steps
295    }

297    #[test]
298    fn detects_start_and_stop() {
299        let mut model = I2cModel::new("i2c_0".to_string(), pins());
300        let mut h = BusHarness::new(2);
301        // Steady state: bus high.
302        let _ = run_steps(&mut model, &mut h, &[(1, 1)]);
303        // START: SDA falls while SCL high
304        let ev1 = run_steps(&mut model, &mut h, &[(0, 1)]);
305        assert_eq!(ev1.len(), 1);
306        assert_eq!(ev1[0].event, "start");
307        // SCL falls (no event), SCL rises (we're mid-transaction, no extra)
308        let _ = run_steps(&mut model, &mut h, &[(0, 0), (1, 0)]);
309        // STOP: bring SCL high then SDA high while SCL high
310        let ev2 = run_steps(&mut model, &mut h, &[(0, 1), (1, 1)]);
311        let stops: Vec<_> = ev2.iter().filter(|e| e.event == "stop").collect();
312        assert_eq!(stops.len(), 1);
313    }

315    #[test]
316    fn emits_address_event_with_rw_bit() {
317        // Write transaction: address 0x50, R/W=0 → byte = 0xA0
318        let mut model = I2cModel::new("i2c_0".to_string(), pins());
319        let mut h = BusHarness::new(2);
320        let _ = run_steps(&mut model, &mut h, &[(1, 1)]);
321        // START
322        let _ = run_steps(&mut model, &mut h, &[(0, 1)]);
323        // SCL low (idle)
324        let _ = run_steps(&mut model, &mut h, &[(0, 0)]);
325        // Shift in 0xA0 = 1010_0000
326        let steps = write_byte_steps(0xA0);
327        let events = run_steps(&mut model, &mut h, &steps);
328        let addrs: Vec<_> = events.iter().filter(|e| e.event == "address").collect();
329        assert_eq!(addrs.len(), 1);
330        assert_eq!(addrs[0].payload, json!(0xA0));
331        // is_read = (0xA0 & 1) = 0 → write transaction
332        assert!(!model.is_read);
333    }

335    #[test]
336    fn emits_write_event_for_data_byte() {
337        let mut model = I2cModel::new("i2c_0".to_string(), pins());
338        let mut h = BusHarness::new(2);
339        // Address (write): 0xA0
340        let _ = run_steps(&mut model, &mut h, &[(1, 1), (0, 1), (0, 0)]);
341        let _ = run_steps(&mut model, &mut h, &write_byte_steps(0xA0));
342        // After: bus = (0, 1), bit_count=8, byte_count=1.
343        // ACK pulse: drop SCL then raise (9th SCL posedge → bit_count rolls to 0),
344        // then drop again to set up for next byte. SDA stays low throughout.
345        let _ = run_steps(&mut model, &mut h, &[(0, 0), (0, 1), (0, 0)]);
346        // Data byte 0x42
347        let events = run_steps(&mut model, &mut h, &write_byte_steps(0x42));
348        let writes: Vec<_> = events.iter().filter(|e| e.event == "write").collect();
349        assert_eq!(writes.len(), 1);
350        assert_eq!(writes[0].payload, json!(0x42));
351    }

353    #[test]
354    fn ack_action_drives_sda_low_during_ack_slot() {
355        let mut model = I2cModel::new("i2c_0".to_string(), pins());
356        let mut h = BusHarness::new(2);
357        model.apply_action(&QueuedAction {
358            event: "ack".to_string(),
359            payload: json!(""),
360        });
361        // START + SCL-low setup, then address byte.
362        let _ = run_steps(&mut model, &mut h, &[(1, 1), (0, 1), (0, 0)]);
363        let _ = run_steps(&mut model, &mut h, &write_byte_steps(0xA0));
364        // After: bus = (0, 1), bit_count=8.
365        // Drop SCL — this is the negedge where the model decides ack vs nack.
366        // Since do_ack=true, drive_sda becomes false (model pulls SDA low).
367        let _ = run_steps(&mut model, &mut h, &[(0, 0)]);
368
369        // Master releases SDA. Verify the model is still pulling it low.
370        let mut overrides = ModelOverrides::new();
371        let mut events = Vec::new();
372        h.set_bit(model.pins.sda_oe_out_pos, 0); // design releases sda → bus sda=1
373                                                 // scl_oe stays at 1 → scl=0
374        model.step(&h.state, &mut overrides, &mut events, 0);
375        // sda_i = sda(=1) AND drive_sda(=false) = 0
376        assert_eq!(overrides[&model.pins.sda_i_in_pos], 0);
377    }
