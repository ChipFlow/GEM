3//! Splitting deep circuit into major stages at global level indices.
4//!
5//! This is crucial in efficiently handling large and deep circuits
6//! with a limited processing element width.
7
8use crate::aig::{DriverType, EndpointGroup, TopoTraverser, AIG};
9use indexmap::IndexSet;
10
11/// A struct representing the boundaries of a staged AIG.
12pub struct StagedAIG {
13    /// the staged primary inputs from previous levels.
14    pub primary_inputs: Option<IndexSet<usize>>,
15    /// the staged primary output pins for next levels.
16    ///
17    /// these pins are active nodes at the level split.
18    pub primary_output_pins: Vec<usize>,
19    /// the endpoint indices of original AIG fulfilled by current level.
20    pub endpoints: Vec<usize>,
21}
22
23impl StagedAIG {
24    /// Get the number of endpoint groups that should be fulfilled
25    /// with this staged AIG.
26    ///
27    /// This mimics the interface given by a raw AIG.
28    pub fn num_endpoint_groups(&self) -> usize {
29        self.primary_output_pins.len() + self.endpoints.len()
30    }
31
32    /// Get the virtual endpoint group with an index.
33    ///
34    /// This mimics the interface given by a raw AIG.
35    pub fn get_endpoint_group<'aig>(&self, aig: &'aig AIG, endpt_id: usize) -> EndpointGroup<'aig> {
36        if endpt_id < self.primary_output_pins.len() {
37            EndpointGroup::StagedIOPin(self.primary_output_pins[endpt_id])
38        } else {
39            aig.get_endpoint_group(self.endpoints[endpt_id - self.primary_output_pins.len()])
40        }
41    }
42
43    /// build a staged AIG that consists of all levels.
44    pub fn from_full_aig(aig: &AIG) -> Self {
45        StagedAIG {
46            primary_inputs: None,
47            primary_output_pins: vec![],
48            endpoints: (0..aig.num_endpoint_groups()).collect(),
49        }
50    }
51
52    /// build a staged AIG by horizontal splitting given a subset
53    /// of endpoints.
54    ///
55    /// return built StagedAIG.
56    /// the endpoints are given as a slice of endpoint group indices,
57    /// that must have all staged primary output groups at the front
58    /// and original endpoints following. otherwise we panic.
59    ///
60    /// the result guarantees that the endpoint `i` corresponds to
61    /// the original staged's endpoint `endpoint_subset[i]`.
62    pub fn to_endpoint_subset(&self, endpoint_subset: &[usize]) -> StagedAIG {
63        let mut staged_sub = StagedAIG {
64            primary_inputs: self.primary_inputs.clone(),
65            primary_output_pins: vec![],
66            endpoints: vec![],
67        };
68        for &endpt_i in endpoint_subset {
69            if endpt_i < self.primary_output_pins.len() {
70                staged_sub
71                    .primary_output_pins
72                    .push(self.primary_output_pins[endpt_i]);
73                assert!(
74                    staged_sub.endpoints.is_empty(),
75                    "endpoint subset must be in order!"
76                );
77            } else {
78                staged_sub
79                    .endpoints
80                    .push(self.endpoints[endpt_i - self.primary_output_pins.len()]);
81            }
82        }
83        staged_sub
84    }
85
86    /// build a staged AIG by vertical splitting at the given level id.
87    ///
88    /// return built StagedAIG.
89    /// the active middle nodes at split can be obtained from the
90    /// StagedAIG::primary_output_pins.
91    /// if this is empty, it means all endpoints are already satisfied
92    /// after this stage.
93    pub fn from_split(
94        aig: &AIG,
95        unrealized_orig_endpoints: &IndexSet<usize>,
96        primary_inputs: Option<&IndexSet<usize>>,
97        split_at_level: usize,
98    ) -> Self {
99        let mut unrealized_endpoint_nodes = Vec::new();
100        for &endpt in unrealized_orig_endpoints {
101            aig.get_endpoint_group(endpt).for_each_input(|i| {
102                unrealized_endpoint_nodes.push(i);
103            });
104        }
105        assert!(!unrealized_endpoint_nodes.is_empty());
106        let mut traverser = TopoTraverser::new(aig.num_aigpins);
107        let order = traverser.topo_traverse(aig, Some(&unrealized_endpoint_nodes), primary_inputs);
108        let mut num_fanouts = vec![0; aig.num_aigpins + 1];
109        let mut level_id = vec![0; aig.num_aigpins + 1];
110        for &i in &order {
111            if matches!(primary_inputs, Some(pi) if pi.contains(&i)) {
112                continue;
113            }
114            if let DriverType::AndGate(a, b) = aig.drivers[i] {
115                if a >= 2 {
116                    num_fanouts[a >> 1] += 1;
117                    level_id[i] = level_id[i].max(level_id[a >> 1] + 1);
118                }
119                if b >= 2 {
120                    num_fanouts[b >> 1] += 1;
121                    level_id[i] = level_id[i].max(level_id[b >> 1] + 1);
122                }
123            }
124        }
125        let mut endpt_level_id = vec![0; aig.num_endpoint_groups()];
126        for &endpt in unrealized_orig_endpoints {
127            aig.get_endpoint_group(endpt).for_each_input(|i| {
128                num_fanouts[i] += 1;
129                endpt_level_id[endpt] = endpt_level_id[endpt].max(level_id[i]);
130            });
131        }
132        let mut nodes_at_split = IndexSet::new();
133        for &i in &order {
134            if level_id[i] > split_at_level {
135                continue;
136            }
137            nodes_at_split.insert(i);
138            if matches!(primary_inputs, Some(pi) if pi.contains(&i)) {
139                continue;
140            }
141            if let DriverType::AndGate(a, b) = aig.drivers[i] {
142                if a >= 2 {
143                    num_fanouts[a >> 1] -= 1;
144                    if num_fanouts[a >> 1] == 0 {
145                        assert!(nodes_at_split.swap_remove(&(a >> 1)));
146                    }
147                }
148                if b >= 2 {
149                    num_fanouts[b >> 1] -= 1;
150                    if num_fanouts[b >> 1] == 0 {
151                        assert!(nodes_at_split.swap_remove(&(b >> 1)));
152                    }
153                }
154            }
155        }
156        let mut endpoints_before_split = Vec::new();
157        for &endpt in unrealized_orig_endpoints {
158            if endpt_level_id[endpt] > split_at_level {
159                continue;
160            }
161            endpoints_before_split.push(endpt);
162            aig.get_endpoint_group(endpt).for_each_input(|i| {
163                num_fanouts[i] -= 1;
164                if num_fanouts[i] == 0 {
165                    assert!(nodes_at_split.swap_remove(&i));
166                }
167            });
168        }
169
170        StagedAIG {
171            primary_inputs: primary_inputs.cloned(),
172            primary_output_pins: nodes_at_split
173                .iter()
174                .copied()
175                .filter(|po| !matches!(primary_inputs, Some(pi) if pi.contains(po)))
176                .collect(),
177            endpoints: endpoints_before_split,
178        }
179    }
180}

182/// Given the level split points, return a list of split stages.
183///
184/// For example, given [10, 20], will return a list like this:
185/// [(0, 10, stage0_10), (10, 20, stage10_20), (20, MAX, stage20_MAX)]
186///
187/// If the netlist ends early before all split points, the length might be
188/// shorter than expected.
189pub fn build_staged_aigs(aig: &AIG, level_split: &[usize]) -> Vec<(usize, usize, StagedAIG)> {
190    let mut ret = Vec::new();
191    let mut unrealized_orig_endpoints = (0..aig.num_endpoint_groups()).collect::<IndexSet<_>>();
192    let mut primary_inputs: Option<IndexSet<usize>> = None;
193
194    for i in 0..level_split.len() {
195        let cur_split = level_split[i];
196        let last_split = match i {
197            0 => 0,
198            i @ _ => level_split[i - 1],
199        };
200        let staged = StagedAIG::from_split(
201            aig,
202            &unrealized_orig_endpoints,
203            primary_inputs.as_ref(),
204            cur_split - last_split,
205        );
206        for &endpt in &staged.endpoints {
207            assert!(unrealized_orig_endpoints.swap_remove(&endpt));
208        }
209        let primary_inputs = primary_inputs.get_or_insert_with(|| Default::default());
210        for &inp in &staged.primary_output_pins {
211            primary_inputs.insert(inp);
212        }
213        if staged.primary_output_pins.is_empty() {
214            ret.push((last_split, usize::MAX, staged));
215            return ret;
216        }
217        ret.push((last_split, cur_split, staged));
218    }
219
220    let last_split = match level_split.len() {
221        0 => 0,
222        i @ _ => level_split[i - 1],
223    };
224    ret.push((
225        last_split,
226        usize::MAX,
227        StagedAIG::from_split(
228            aig,
229            &unrealized_orig_endpoints,
230            primary_inputs.as_ref(),
231            usize::MAX,
232        ),
233    ));
234
235    ret
236}
